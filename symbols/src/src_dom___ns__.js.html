 95 /**
 96  * Alias of W3C <code>element.getElementsByClassName</code>.
 97  * Used to reduce size after compilation.
 98  * @param {!Element|Node} element Element to start searching.
 99  * @param {string} className Class name to match.
100  * @return {!Array|NodeList} Array of found elements.
101  * @static
102  */
103 dom.getElementsByClassName = function(element, className) {
104   if (element.getElementsByClassName) {
105     return element.getElementsByClassName(className);
106   }
107 
108   if (element.querySelectorAll) {
109     return element.querySelectorAll('.' + className);
110   }
111 
112   /** @type {!RegExp} */
113   var re = new RegExp('(?:^|\\s)' + className + '(?!\\S)');
114   /** @type {!Array.<Node>} */ var result = [];
115   /** @type {NodeList} */ var nodes = element.getElementsByTagName('*');
116   for (/** @type {number} */ var i = 0; i < nodes.length; i++) {
117     if (re.test(nodes[i].className)) {
118       result.push(nodes[i]);
119     }
120   }
121   return result;
122 };

125 /**
126  * Alias of W3C <code>element.getBoundingClientRect</code>.
127  * Used to reduce size after compilation.
128  * @param {!Element|Node} element Element for calculating bounding rect.
129  * @return {Object} Returns dict {top, left, width, height, right, bottom}.
130  */
131 dom.getBoundingClientRect = function(element) {
132   /** @type {Object} */
133   var rect = element.getBoundingClientRect && element.getBoundingClientRect();
134   if (!rect) {
135     rect = {'top': 0, 'left': 0, 'width': element.offsetWidth,
136       'height': element.offsetHeight};
137     while (element && !isNaN(element.offsetLeft) && !isNaN(element.offsetTop)) {
138       rect['left'] += element.offsetLeft - element.scrollLeft;
139       rect['top'] += element.offsetTop - element.scrollTop;
140       element = element.offsetParent;
141     }
142     rect['right'] = rect['left'] + rect['width'];
143     rect['bottom'] = rect['top'] + rect['height'];
144   }
145   return rect;
146 };

149 /**
150  * Alias of W3C <code>document.defaultView.getComputedStyle</code>.
151  * Used to reduce size after compilation.
152  * @param {!Element|Node} element Element for getting style.
153  * @param {string} prop Style property name.
154  * @return {string|number} Returns element style value.
155  */
156 dom.getComputedStyle = function(element, prop) {
157   /** @type {string} */ var value = element.style[prop];
158   if (element.currentStyle)
159     value = element.currentStyle[prop];
160   else if (window.getComputedStyle)
161     value = dom.document.defaultView.getComputedStyle(
162         element, null).getPropertyValue(prop);
163   return value;
164 };

167 /**
168  * Clears element content.
169  * @param {!Element|Node} element Element to clear.
170  */
171 dom.clearElement = function(element) {
172   switch (element.tagName.toUpperCase()) {
173     case 'TABLE':
174     case 'THEAD':
175     case 'TBODY':
176     case 'TFOOT':
177     case 'TR':
178     case 'SELECT':
179       while (element.lastChild) {
180         element.removeChild(element.lastChild);
181       }
182       break;
183     default:
184       element.innerHTML = '';
185   }
186 };

189 /**
190  * Returns true if an element has a class.
191  * @param {Node} element DOM node to test.
192  * @param {string} className Class name to test for.
193  * @return {boolean} Whether element has the class.
194  */
195 dom.css.hasClass = function(element, className) {
196   // Note: `element.classList.contains` throws 'TypeError: Illegal invocation'
197   // if element is not appended into the DOM.
198   if (element) {
199     /** @type {Array.<string>} */ var classes = element.className.split(' ');
200     for (/** @type {number} */ var i = 0; i < classes.length;) {
201       if (className == classes[i++]) {
202         return true;
203       }
204     }
205   }
206   return false;
207 };

222 /**
223  * Adds a class or classes to an element. Does not add multiples of class names.
224  * @param {Node} element DOM node to add class to.
225  * @param {...string} var_args Class names to add.
226  */
227 dom.css.addClass = function(element, var_args) {
228   // Note: `element.classList.add` throws 'TypeError: Illegal invocation'
229   // if element is not appended into the DOM.
230   if (element) {
231     dom.css.removeClass.apply(null, arguments);
232 
233     /** @type {Array.<string>} */
234     var args = Array.prototype.slice.call(arguments, 1);
235     element.className += ' ' + args.join(' ');
236   }
237 };

240 /**
241  * Removes a class or classes from an element.
242  * @param {Node} element DOM node to remove class from.
243  * @param {...string} var_args Class name(s) to remove.
244  */
245 dom.css.removeClass = function(element, var_args) {
246   // Note: `element.classList.remove` throws 'TypeError: Illegal invocation'
247   // if element is not appended into the DOM.
248   if (element) {
249     /** @type {Array.<string>} */
250     var args = Array.prototype.slice.call(arguments, 1);
251     dom.css.setClass(element, element.className.replace(
252         new RegExp('\\s?\\b(' + args.join('|') + ')\\b', 'g'), ''));
253   }
254 };

324 /**
325  * Alias of W3C <code>element.dispatchEvent</code>.
326  * Used to reduce size after compilation.
327  * @param {Element|Node|Window} element Element to dispatch event.
328  * @param {string} type Type of event.
329  * @return {boolean} Returns <code>true</code> if event dispatched successfully.
330  * @static
331  */
332 dom.events.dispatchEvent = function(element, type) {
333   /** @type {Event} */ var evt = null;
334   /** @type {boolean} */ var result = false;
335 
336   if (dom.document.createEvent) {
337     evt = dom.document.createEvent('HTMLEvents');
338     // initEvent(type, bubbling, cancelable)
339     evt.initEvent(type, true, true);
340     result = !element.dispatchEvent(evt);
341   } else if (dom.document.createEventObject) {
342     evt = dom.document.createEventObject();
343     try {
344       // MSIE throws error on invalid event type.
345       result = element.fireEvent('on' + type, evt);
346     } catch (e) {}
347   }
348   return result;
349 };
