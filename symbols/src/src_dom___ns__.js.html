101 /**
102  * Alias of W3C <code>element.getElementsByClassName</code>.
103  * Used to reduce size after compilation.
104  * @param {!Element|Node} element Element to start searching.
105  * @param {string} className Class name to match.
106  * @return {!Array.<Node>|NodeList} Array of found elements.
107  * @static
108  */
109 dom.getElementsByClassName = function(element, className) {
110   if (element.getElementsByClassName) {
111     return element.getElementsByClassName(className);
112   }
113 
114   if (element.querySelectorAll) {
115     return element.querySelectorAll('.' + className);
116   }
117 
118   /** @type {!RegExp} */
119   var re = new RegExp('(?:^|\\s)' + className + '(?!\\S)');
120   /** @type {!Array.<Node>} */ var result = [];
121   /** @type {NodeList} */ var nodes = dom.getElementsByTagName(element, '*');
122   for (/** @type {number} */ var i = 0; i < nodes.length; i++) {
123     if (re.test(nodes[i].className)) {
124       result.push(nodes[i]);
125     }
126   }
127   return result;
128 };

131 /**
132  * Alias of W3C <code>element.getBoundingClientRect</code>.
133  * Used to reduce size after compilation.
134  * @param {!Element|Node} element Element for calculating bounding rect.
135  * @return {Object} Returns dict {top, left, width, height, right, bottom}.
136  */
137 dom.getBoundingClientRect = function(element) {
138   /** @type {Object} */
139   var rect = element.getBoundingClientRect && element.getBoundingClientRect();
140   if (!rect) {
141     rect = {'top': 0, 'left': 0, 'width': element.offsetWidth,
142       'height': element.offsetHeight};
143     while (element && !isNaN(element.offsetLeft) && !isNaN(element.offsetTop)) {
144       rect['left'] += element.offsetLeft - element.scrollLeft;
145       rect['top'] += element.offsetTop - element.scrollTop;
146       element = element.offsetParent;
147     }
148     rect['right'] = rect['left'] + rect['width'];
149     rect['bottom'] = rect['top'] + rect['height'];
150   }
151   return rect;
152 };

156  * Alias of W3C <code>document.defaultView.getComputedStyle</code>.
157  * Used to reduce size after compilation.
158  * @param {!Element|Node} element Element for getting style.
159  * @param {string} prop Style property name.
160  * @return {string|number} Returns element style value.
161  */
162 dom.getComputedStyle = function(element, prop) {
163   /** @type {string} */ var value = element.style[prop];
164   if (element.currentStyle)
165     value = element.currentStyle[prop];
166   else if (window.getComputedStyle)
167     value = dom.document.defaultView.getComputedStyle(
168         element, dom.NULL).getPropertyValue(prop);
169   return value;
170 };

173 /**
174  * Clears element content.
175  * @param {!Element|Node} element Element to clear.
176  */
177 dom.clearElement = function(element) {
178   switch (element.tagName.toUpperCase()) {
179     case 'TABLE':
180     case 'THEAD':
181     case 'TBODY':
182     case 'TFOOT':
183     case 'TR':
184     case 'SELECT':
185       while (element.lastChild) {
186         element.removeChild(element.lastChild);
187       }
188       break;
189     default:
190       element.innerHTML = '';
191   }
192 };

195 /**
196  * Returns true if an element has a class.
197  * @param {Node} element The element to test.
198  * @param {string} className The class name to test for.
199  * @return {boolean} Whether element has the class.
200  */
201 dom.css.hasClass = function(element, className) {
202   // Note: `element.classList.contains` throws 'TypeError: Illegal invocation'
203   // if element is not appended into the DOM.
204   if (element) {
205     /** @type {Array.<string>} */ var classes = element.className.split(' ');
206     for (/** @type {number} */ var i = 0; i < classes.length;) {
207       if (className == classes[i++]) {
208         return true;
209       }
210     }
211   }
212   return false;
213 };

216 /**
217  * Sets the entire class name of an element.
218  * @param {Node} element The element to set class of.
219  * @param {...string} var_args The class name(s) to apply to element.
220  */
221 dom.css.setClass = function(element, var_args) {
222   if (element) {
223     /** @type {!Array.<string>} */ var args = dom.css.slice_.call(arguments, 1);
224     element.className = args.join(' ');
225   }
226 };

229 /**
230  * Adds a class or classes to an element. Does not add multiples of class names.
231  * @param {Node} element The element to add class to.
232  * @param {...string} var_args The class name(s) to add.
233  */
234 dom.css.addClass = function(element, var_args) {
235   // Note: `element.classList.add` throws 'TypeError: Illegal invocation'
236   // if element is not appended into the DOM.
237   if (element) {
238     dom.css.removeClass.apply(dom.NULL, arguments);
239 
240     /** @type {!Array.<string>} */ var args = dom.css.slice_.call(arguments, 1);
241     element.className += ' ' + args.join(' ');
242   }
243 };

245 
246 /**
247  * Removes a class or classes from an element.
248  * @param {Node} element The element to remove class from.
249  * @param {...string} var_args The class name(s) to remove.
250  */
251 dom.css.removeClass = function(element, var_args) {
252   // Note: `element.classList.remove` throws 'TypeError: Illegal invocation'
253   // if element is not appended into the DOM.
254   if (element) {
255     /** @type {!Array.<string>} */ var args = dom.css.slice_.call(arguments, 1);
256     dom.css.setClass(element, element.className.replace(
257         new RegExp('\\s?\\b(' + args.join('|') + ')\\b', 'g'), ''));
258   }
259 };

262 /**
263  * Toggles element class name.
264  * @param {Node} element The element to add or remove the class on.
265  * @param {string} className The class name to toggle.
266  */
267 dom.css.toggleClass = function(element, className) {
268   (dom.css.hasClass(element, className) ?
269       dom.css.removeClass : dom.css.addClass)(element, className);
270 };

352 /**
353  * Alias of W3C <code>element.dispatchEvent</code>.
354  * Used to reduce size after compilation.
355  * @param {Element|Node|Window} element Element to dispatch event.
356  * @param {string} type Type of event.
357  * @return {boolean} Returns <code>true</code> if event dispatched successfully.
358  * @static
359  */
360 dom.events.dispatchEvent = function(element, type) {
361   /** @type {Event} */ var evt = dom.NULL;
362   /** @type {boolean} */ var result = false;
363 
364   if (dom.document.createEvent) {
365     evt = dom.document.createEvent('HTMLEvents');
366     // initEvent(type, bubbling, cancelable)
367     evt.initEvent(type, true, true);
368     result = !element.dispatchEvent(evt);
369   } else if (dom.document.createEventObject) {
370     evt = dom.document.createEventObject();
371     try {
372       // MSIE throws error on invalid event type.
373       result = element.fireEvent('on' + type, evt);
374     } catch (e) {}
375   }
376   return result;
377 };
