 10 /**
 11  * Constructor of Calendar.
 12  * @param {string|Node} container The HTML container or its ID.
 13  * @param {Object=} opt_options Optional options.
 14  * @extends {dom.EventDispatcher} dom.EventDispatcher
 15  * @constructor
 16  * @requires formatters.DateFormatter
 17  * @requires locale.Calendar
 18  * @example
 19  * <style>
 20  * table.calendar {border: solid 1px gray; border-collapse: collapse;}
 21  * table.calendar thead {background: silver;}
 22  * table.calendar td,
 23  * table.calendar th {border: solid 1px gray;}
 24  * table.calendar td.today {font-weight: bold; background: silver;}
 25  * table.calendar td.active {font-weight: bold; background: blue; color: white;}
 26  * table.calendar td.disabled {color: gray;}
 27  * table.calendar th.prev,
 28  * table.calendar th.next {cursor: pointer;}
 29  * </style>
 30  * <div id="calendar-container"></div>
 31  * <script>controls.Calendar.draw('calendar-container');</script>
 32  * <script>
 33  * var cal = new controls.Calendar('calendar-container', {
 34  *   'format': 'YYYY-MM-dd', // Default date format.
 35  *   'selectable': false,    // Specifies if date is selectable.
 36  *   'multiple': false,      // Allows multiple selection.
 37  *   'empty-row': false      // Display an empty row for short month.
 38  * });
 39  * cal.draw();
 40  * </script>
 41  * <style>
 42  * table.calendar {border: solid 1px gray; border-collapse: collapse;
 43  *                 background: white; font-family: Arial; font-size: 13px;}
 44  * table.calendar thead {background: silver;}
 45  * table.calendar td,
 46  * table.calendar th {border: solid 1px gray;}
 47  * table.calendar td.today {font-weight: bold; background: silver;}
 48  * table.calendar td.active {font-weight: bold; background: blue; color: white;}
 49  * table.calendar td.disabled {color: gray;}
 50  * table.calendar th.prev,
 51  * table.calendar th.next {cursor: pointer;}
 52  * </style>
 53  * <script src="../../bin/jscb.js"></script>
 54  * <div id="calendar-container"></div>
 55  * <script>controls.Calendar.draw('calendar-container')</script>

 57 controls.Calendar = function(container, opt_options) {
 58   dom.EventDispatcher.apply(this, arguments);
 59 
 60   opt_options = opt_options || {};
 61   opt_options['format'] = opt_options['format'] || 'YYYY-MM-dd';
 62   //opt_options['selectable'] = opt_options['selectable'] || true;
 63   //opt_options['multiple'] = opt_options['multiple'] || true;
 64   //opt_options['empty-row'] = opt_options['empty-row'] || true;
 65 
 66   /**
 67    * Enumeration of calendar events.
 68    * @enum {string}
 69    * @example <code>{
 70    *  PREV_MONTH, NEXT_MONTH, DRAW
 71    * }</code>
 72    */
 73   this.events = {
 74     NEXT_MONTH: 'next.month',
 75     PREV_MONTH: 'prev.month',
 76     DRAW: 'draw'
 77   };
 78 
 79   /**
 80    * Draws calendar into specified <code>container</code>.
 81    * Dispatches <code>controls.Calendar.events.DRAW</code> event and
 82    * dispatches <code>dom.events.TYPE.CLICK</code> event when clicking on cell.
 83    * Exported for closure compiler with the same name.
 84    * @param {Array.<Date>|Date=} opt_selected Optional list of selected dates.
 85    */
 86   this.draw = function(opt_selected) {
 87     initSelection_(opt_selected);
 88 
 89     container_.innerHTML = '<table class="calendar skiptranslate">' +
 90                            thead_(navDate_) +
 91                            tbody_(navDate_) + '</table>';
 92     /** @type {!HTMLTableElement} */ var table = getTable_();
 93 
 94     if (!opt_options['empty-row']) {
 95       removeEmptyRow_(table);
 96     }
 97 
 98     initEvents_(table);
 99     self_.dispatchEvent(self_.events.DRAW);
100   };
101 
102   // Export for closure compiler.
103   this['draw'] = this.draw;

139   /**
140    * Iterates over table cells.
141    * @param {!function(Element)} callback Iterator callback function.
142    */
143   this.each = function(callback) {
144     /** @type {Array|NodeList} */
145     var cells = dom.getElementsByTagName(getTable_(), 'TD') || [];
146     for (/** @type {number} */ var i = 0; i < cells.length;) {
147       callback(cells[i++]);
148     }
149   };

151   /**
152    * Clears today, selected and active days.
153    */
154   this.clear = function() {
155     selected_ = [];
156     var cellDate;
157     var now = formatter_.format(new Date(), 'YYYY-MM-dd');
158     self_.each(function(cell) {
159       cellDate = cell.getAttribute('value');
160       dom.css.setClass(cell, cellDate > now ? 'disabled' : cellDate == now ?
161           'today' : '');
162     });
163   };

165   /**
166    * @return {!HTMLTableElement} Returns reference to table element.
167    * @private
168    */
169   function getTable_() {
170     return dom.getElementsByTagName(container_, 'TABLE')[0];
171   }

173   /**
174    * @param {!HTMLTableElement} table The table element.
175    * @private
176    */
177   function removeEmptyRow_(table) {
178     /** @type {HTMLCollection} */ var rows = table.rows;
179     /** @type {HTMLTableRowElement} */ var row = rows[rows.length - 1];
180     /** @type {number} */ var content = 0;
181     for (/** @type {number} */ var i = 0; i < row.cells.length;) {
182       content += +row.cells[i++].innerHTML || 0;
183     }
184 
185     if (!content) {
186       table.deleteRow(table.rows.length - 1);
187     }
188   }

190   /**
191    * @param {Array.<Date>|Date=} opt_selected Optional list of selected dates.
192    * @private
193    */
194   function initSelection_(opt_selected) {
195     opt_selected = opt_selected || new Date;
196     if (!(opt_selected instanceof Array)) {
197       opt_selected = [opt_selected];
198     }
199 
200     navDate_ = navDate_ || new Date(opt_selected[0].getTime());
201     selected_[0] = selected_[0] || opt_selected[0];
202     if (opt_options['multiple']) {
203       for (/** @type {number} */ var i = 1; i < opt_selected.length; i++) {
204         selected_[i] = opt_selected[i];
205       }
206     }
207   }

209   /**
210    * @param {!HTMLTableElement} table The table element.
211    * @private
212    */
213   function initEvents_(table) {
214     /** @type {!HTMLCollection} */ var headers = table.rows[0].cells;
215     /** @type {Array.<Node>} */ var cells = [
216       headers[0],
217       headers[headers.length - 1]
218     ];
219     for (/** @type {number} */ var i = 0; i < cells.length; i++) {
220       dom.events.addEventListener(cells[i], dom.events.TYPE.CLICK, function(e) {
221         e = e || window.event;
222         self_.dispatchEvent((e.target || e.srcElement).cellIndex ?
223             self_.events.NEXT_MONTH : self_.events.PREV_MONTH);
224       });
225     }
226 
227     if (opt_options['selectable']) {
228       self_.each(function(cell) {
229         dom.events.addEventListener(cell, dom.events.TYPE.CLICK, clickHandler_);
230       });
231     }
232   }
233 
234   /**
235    * @param {Event} e The event.
236    * @private
237    */
238   function clickHandler_(e) {
239     e = e || window.event;
240     /** @type {Element} */ var target = e.target || e.srcElement;
241     if (!isNaN(+target.innerHTML)) {
242       /** @type {Date} */ var selected = formatter_.parse(
243           target.getAttribute('value'), opt_options['format']);
244 
245       if (opt_options['multiple'] && !first_) {
246         selected_.push(selected);
247         /** @type {number} */ var index = selected_.length - 1;
248         if (selected_[0] > selected_[index]) {
249           selected_.reverse();
250         }
251 
252         /** @type {Date} */ var start = selected_[0];
253         /** @type {Date} */ var end = new Date(selected_[index]);
254         while (end > start) {
255           end.setDate(end.getDate() - 1);
256           selected_.splice(1, 0, new Date(end.getTime()));
257         }
258 
259         self_.draw(selected_);
260       } else {
261         self_.clear();
262         dom.css.setClass(target, 'active');
263         selected_[0] = selected;
264       }
265       first_ = !first_;
266       self_.dispatchEvent(dom.events.TYPE.CLICK);
267     }
268   }

270   /**
271    * @param {number} index Navigation direction index.
272    * @private
273    */
274   function navigate_(index) {
275     navDate_.setDate(1);
276     navDate_.setMonth(navDate_.getMonth() + (index ? 1 : -1));
277     self_.draw();
278   }

280   /**
281    * @param {!Date} date The date object.
282    * @return {string} Returns <TBODY> html markup.
283    * @private
284    */
285   function tbody_(date) {
286     /** @type {!Date} */ var now = new Date;
287     /** @type {number} */ var year = date.getFullYear();
288     /** @type {number} */ var month = date.getMonth();
289 
290     // Fix date bug when current day is 31st.
291     /** @type {number} */ var fix = new Date(year, month, 1).getDay() + 1;
292 
293     /** @type {Array} */
294     var dim = [31, 0, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];
295     dim[1] = (((year % 100 != 0) && (year % 4 == 0)) ||
296         (year % 400 == 0)) ? 29 : 28;
297 
298     /** @type {string} */ var tbody = '<tbody><tr>';
299     for (/** @type {number} */ var i = 1; i <= 42; i++) {
300       /** @type {number} */
301       var day = ((i - fix >= 0) && (i - fix < dim[month])) ? i - fix + 1 : 0;
302       /** @type {string} */
303       var value = !day ? '' : formatter_.format(
304           new Date(year, month, day), opt_options['format']);
305 
306       /** @type {string} */ var className = getCellCssClass_(date, now, day);
307       tbody += '<td align="right" class="' + className + '"' +
308                ' value="' + value + '">' + (day || ' ') + '</td>';
309       if (((i) % 7 == 0) && (i < 36)) tbody += '</tr><tr>';
310     }
311     tbody += '</tr></tbody>';
312     return tbody;
313   }

315   /**
316    * @param {!Date} date The date object.
317    * @return {string} Returns <THEAD> HTML markup.
318    * @private
319    */
320   function thead_(date) {
321     return '<thead><tr><th class=prev><</th><th colspan=5>' +
322            locale_.getMonthName(date) + ' ' + date.getFullYear() +
323            '</th><th class=next>></th></tr><tr><th>' +
324            (locale_.getWeekNames().join('</th><th>')) + '</th></tr></thead>';
325   }

327   /**
328    * @param {!Date} date The calendar date.
329    * @param {!Date} now The current date.
330    * @param {number} day The calendar day.
331    * @return {string} Returns cell CSS class for calendar <code>day</code>.
332    * @private
333    */
334   function getCellCssClass_(date, now, day) {
335     /** @type {number} */ var year = date.getFullYear();
336     /** @type {number} */ var month = date.getMonth();
337     /** @type {number} */ var active = 0;
338     /** @type {boolean} */ var disabled = new Date(year, month, day) > now;
339     /** @type {number} */
340     var today = (year == now.getFullYear() &&
341                  month == now.getMonth()) ? now.getDate() : 0;
342 
343     for (/** @type {number} */ var i = 0; i < selected_.length; i++) {
344       /** @type {Date} */ var selected = selected_[i];
345       if (year == selected.getFullYear() &&
346           month == selected.getMonth() &&
347           day == selected.getDate()) {
348         active = day;
349         break;
350       }
351     }
352 
353     return day ? (day == today ? 'today' :
354         (active ? 'active' :
355         (disabled ? 'disabled' : ''))) : '';
356   }

363     self_.addEventListener(self_.events.NEXT_MONTH, function() {navigate_(1);});
364   }
