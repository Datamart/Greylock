 17 /**
 18  * @param {string} data Data content.
 19  * @param {string=} opt_fileName Optional file name.
 20  * @param {string=} opt_contentType Optional content type.
 21  * @static
 22  */
 23 util.FileUtils.saveAs = function(data, opt_fileName, opt_contentType) {
 24   /** @type {Element} */ var link = dom.createElement('A');
 25   opt_fileName = opt_fileName || 'file';
 26   opt_contentType = opt_contentType || 'text/plain';
 27   if (window.Blob && 'download' in link) {
 28     link['href'] = (window.URL || window.webkitURL).createObjectURL(
 29         new Blob([data],
 30         {'type': opt_contentType + ';charset=' + dom.CHARSET}));
 31     link.setAttribute('download', opt_fileName);
 32     dom.document.body.appendChild(link);
 33     link.click();
 34   } else {
 35     // @link http://msdn.microsoft.com/en-us/library/cc848897(VS.85).aspx
 36     window.open('data:' + opt_contentType +
 37                 ';base64,' + util.StringUtils.Base64.encode(data));
 38   }
 39 };
