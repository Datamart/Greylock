 16 /**
 17  * Extends source with target options.
 18  * @param {!Object} target Options map.
 19  * @param {!Object} source Options map.
 20  * @return {!Object} A map of key/value pairs of options.
 21  * @static
 22  */
 23 util.Object.extend = function(target, source) {
 24   for (/** @type {string} */ var key in source) {
 25     if (source[key] instanceof Array) {
 26       target[key] = [].concat(source[key]);
 27     } else if ('object' == typeof source[key]) {
 28       target[key] = util.Object.extend(
 29           /** @type {!Object} */ (target[key] || {}),
 30           /** @type {!Object} */ (source[key]));
 31     } else {
 32       target[key] = source[key];
 33     }
 34   }
 35   return /** @type {!Object} */ (target);
 36 };

 39 /**
 40  * Returns list of object keys.
 41  * @param {!Object} obj Target Object.
 42  * @return {!Array.<string>} List of target object keys.
 43  * @static
 44  */
 45 util.Object.keys = Object.keys || function(obj) {
 46   /** @type {!Array} */ var keys = [];
 47   /** @type {number} */ var i = 0;
 48   for (/** @type {string|number} */ var key in obj) {
 49     keys[i++] = key;
 50   }
 51   return keys;
 52 };

 55 /**
 56  * Filters object.
 57  * @param {!Object} obj The target object.
 58  * @param {!function(*): boolean} func The filter function.
 59  * @return {!Object} Return filtered object.
 60  * @static
 61  */
 62 util.Object.filter = function(obj, func) {
 63   /** @type {!Object} */ var result = {};
 64   for (/** @type {string|number} */ var key in obj) {
 65     if (func(obj[key])) {
 66       result[key] = obj[key];
 67     }
 68   }
 69   return result;
 70 };
