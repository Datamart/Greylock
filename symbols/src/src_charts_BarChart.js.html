 11 /**
 12  * BarChart constructor.
 13  * @param {string|Element} container The HTML container.
 14  * @constructor
 15  * @class Simple horizontal bar chart implementation.
 16  * @extends {charts.BaseChart} charts.BaseChart
 17  * @requires animation
 18  * @requires charts.Grid
 19  * @requires formatters.NumberFormatter
 20  * @example
 21  * <b>var</b> chart = <b>new</b> charts.BarChart('container_id');
 22  * chart.draw([['Year', 'Sales', 'Expenses', 'Profit'],
 23  *             [2011, 80, 30, 45], [2012, 65, 130, 90], [2013, 45, 100, 60]]);
 24  *
 25  * <div style="border: solid 1px #ccc; margin: 5px; padding: 5px; width: 560px">
 26  *   <div id="chart-container"
 27  *        style="width: 560px; height: 200px;"></div>
 28  * </div>
 29  * <script src="../bin/jscb.js"></script>
 30  * <script>
 31  *   var chart = new charts.BarChart('chart-container');
 32  *   chart.draw([['Year', 'Sales', 'Expenses', 'Profit'],
 33  *               [2011, 80, 30, 45], [2012, 65, 130, 90], [2013, 45, 100, 60]]);
 34  * </script>
 35  */
 36 charts.BarChart = function(container) {
 37   charts.BaseChart.apply(this, arguments);
 38 
 39   /**
 40    * Draws the chart based on <code>data</code> and <code>opt_options</code>.
 41    * @param {!Array.<Array>} data A chart data.
 42    * @param {Object=} opt_options A optional chart's configuration options.
 43    * @override
 44    * @see charts.BaseChart#getOptions
 45    * @example
 46    * options: {
 47    *   'stroke': 1,
 48    *   'font': {'size': 11},
 49    *   'flip': <b>false</b>
 50    * }
 51    */
 52   this.draw = function(data, opt_options) {
 53     data_ = data;
 54     options_ = getOptions_(opt_options);
 55     options_['direction'] = getDirection_();
 56     self_.tooltip.setOptions(options_);
 57 
 58     /** @type {string} */ var content = '';
 59     /** @type {!Array.<Array>} */ var rows = self_.getDataRows(data);
 60     /** @type {!Array.<string>} */ var columns = self_.getDataColumns(data);
 61     /** @type {!Array.<number>} */ var range = self_.getDataRange(data, 1);
 62 
 63     /** @type {number} */ var width = self_.container.offsetWidth || 200;
 64     /** @type {number} */ var height = self_.container.offsetHeight || width;
 65     /** @type {number} */
 66     var border = /** @type {number} */ (options_['stroke']);
 67     for (/** @type {number} */ var i = 0; i < rows.length; i++) {
 68       bars_ += rows[i].length;
 69     }
 70 
 71     barWidth_ = width / bars_ - border * 2;
 72     barHeight_ = height / bars_ - border * 2;
 73 
 74     bars_ = 0;
 75     for (i = 0; i < rows.length; i++) {
 76       /** @type {Array} */ var row = rows[i];
 77       for (/** @type {number} */ var j = 1; j < row.length; j++) {
 78         /** @type {string} */ var tooltip = self_.tooltip.parse(
 79             row[0], columns[j], formatter_.format(row[j]));
 80 
 81         content += getBarContent_(getBarRect_(rows, i, j, range),
 82                                   options_['colors'][j], tooltip);
 83         bars_++;
 84       }
 85 
 86       bars_++; // Add space between bar groups.
 87     }
 88 
 89     initGrid_(range);
 90     self_.drawContent(content);
 91     initEvents_();
 92   };
 93 
 94   // Export for closure compiler.
 95   this['draw'] = this.draw;
 96 
 97   /**
 98    * @param {!Array.<number>} range The data range with min and max values.
 99    * @private
100    */
101   function initGrid_(range) {
102     /** @type {number} */ var minValue = range[0];
103     /** @type {number} */ var maxValue = range[1];
104     /** @type {!Array.<string>} */ var columns = [];
105     //for (/** @type {number} */ var i = 0; i < data_.length; i++) {
106     //columns.push('');
107     //columns.push(data_[i][0]);
108     //}
109     /** @type {!charts.Grid} */ var grid = new charts.Grid(self_.container);
110     options_['data'] = {'min': minValue, 'max': maxValue, 'columns': columns};
111     options_['padding'] = 0;
112     grid.draw(options_);
113   }
114 
115   /**
116    * Gets chart's direction.
117    * @return {number} Returns chart direction.
118    * @see charts.BarChart#DIRECTION
119    * @private
120    */
121   function getDirection_() {
122     /** @type {number} */ var direction = options_['direction'] ||
123                                           charts.Grid.DIRECTION.LEFT_TO_RIGHT;
124     if (options_['flip']) {
125       if (direction == charts.Grid.DIRECTION.LEFT_TO_RIGHT)
126         direction = charts.Grid.DIRECTION.RIGHT_TO_LEFT;
127       else if (direction == charts.Grid.DIRECTION.BOTTOM_TO_TOP)
128         direction = charts.Grid.DIRECTION.TOP_TO_BOTTOM;
129     }
130     return direction;
131   }
132 
133   /**
134    * @param {!Array.<Array>} rows Chart data rows.
135    * @param {number} rowIndex Current row index.
136    * @param {number} columnIndex Current column index.
137    * @param {!Array.<number>} range The data range with min and max values.
138    * @return {!Object.<string, number>} Returns rect as {x, y, width, height}.
139    * @private
140    */
141   function getBarRect_(rows, rowIndex, columnIndex, range) {
142     /** @type {!Object.<string, number>} */
143     var rect = {x: 0, y: 0, width: 0, height: 0};
144 
145     /** @type {number} */ var width = self_.container.offsetWidth || 200;
146     /** @type {number} */ var height = self_.container.offsetHeight || width;
147     /** @type {number} */
148     var border = /** @type {number} */ (options_['stroke']);
149 
150     /** @type {number} */ var minValue = range[0];
151     /** @type {number} */ var maxValue = range[1];
152 
153     /** @type {Array} */ var row = rows[rowIndex];
154     /** @type {number} */ var value = row[columnIndex];
155     /** @type {number} */ var direction = getDirection_();
156 
157     if (direction < charts.Grid.DIRECTION.BOTTOM_TO_TOP) {
158       // Horizontal charts.
159       rect.width = value / maxValue * (width - border * 4);
160       // rect.height = (height / row.length / rows.length) - border * 2;
161       rect.height = barHeight_;
162       rect.x = border;
163       rect.y = (rect.height + border * 2) * bars_ + (rect.height + border * 2);
164       //if (charts.Grid.DIRECTION.RIGHT_TO_LEFT == direction)
165       //  rect.x = width - rect.width - border;
166     } else if (direction >= charts.Grid.DIRECTION.BOTTOM_TO_TOP) {
167       // Vertical charts.
168       // rect.width = (width / row.length / rows.length) - border * 2;
169       rect.width = barWidth_;
170       rect.height = value / maxValue * (height - border * 4);
171       rect.x = (rect.width + border * 2) * bars_ + (rect.width + border * 2);
172       rect.y = border;
173       //if (charts.Grid.DIRECTION.BOTTOM_TO_TOP == direction)
174       //  rect.y = height - rect.height;
175     }
176 
177     return rect;
178   }

180   /**
181    * Gets chart's options merged with defaults chart's options.
182    * @param {Object=} opt_options A optional chart's configuration options.
183    * @return {!Object.<string, *>} A map of name/value pairs.
184    * @see charts.BaseChart#getOptions
185    * @private
186    */
187   function getOptions_(opt_options) {
188     opt_options = opt_options || {};
189     opt_options['stroke'] = opt_options['stroke'] || 1;
190     opt_options['font'] = opt_options['font'] || {};
191     opt_options['font']['size'] = opt_options['font']['size'] || 11;
192     return self_.getOptions(opt_options);
193   }

195   /**
196    * @param {!Object.<string, number>} rect Bar rect.
197    * @param {string} color Bar color.
198    * @param {string} tooltip Bar tooltip.
199    * @return {string} Returns bar content as HTML markup string.
200    * @private
201    */
202   function getBarContent_(rect, color, tooltip) {
203     return '<div class="bar" style="' +
204         'width:' + rect.width + 'px;' +
205         'height:' + rect.height + 'px;' +
206         'top:' + rect.y + 'px;' +
207         'left:' + rect.x + 'px;' +
208         'background:' + color + ';' +
209         'border:solid ' + options_['stroke'] + 'px #fff;' +
210         'color:#fff;' +
211         'overflow:hidden;' +
212         'position:absolute;' +
213         'opacity:' + options_['opacity'] + ';' +
214         'filter: alpha(opacity=' + (options_['opacity'] * 100) + ');' +
215         'font-family:' + options_['font']['family'] + ';' +
216         'font-size:' + options_['font']['size'] + 'px;' +
217         '" tooltip="' + tooltip + '"></div>';
218   }

220   /**
221    * Initializes events handlers.
222    * @private
223    */
224   function initEvents_() {
225     /** @type {!Array|NodeList} */
226     var nodes = dom.getElementsByClassName(self_.container, 'bar');
227     for (/** @type {number} */ var i = 0; i < nodes.length; i++) {
228       setEvents_(nodes[i]);
229     }
230   }
231 
232   /**
233    * Sets events handlers.
234    * @param {!Element} node The element.
235    * @private
236    */
237   function setEvents_(node) {
238     dom.events.addEventListener(node, dom.events.TYPE.MOUSEOVER, function(e) {
239       node.style.opacity = 1;
240       node.style.filter = 'alpha(opacity=100)';
241 
242       self_.tooltip.show(e);
243       /** @type {!Object.<string, number>} */
244       var position = getTooltipPosition_(node);
245       self_.tooltip.show(e, position.x, position.y);
246     });
247 
248     dom.events.addEventListener(node, dom.events.TYPE.MOUSEOUT, function(e) {
249       node.style.opacity = options_['opacity'];
250       node.style.filter = 'alpha(opacity=' + (options_['opacity'] * 100) + ')';
251       self_.tooltip.hide(e);
252     });
253 
254     dom.events.dispatchEvent(node, dom.events.TYPE.MOUSEOUT);
255     initAnimation_(node);
256   }
257 
258   /**
259    * Initializes bar animation.
260    * @param {!Element|Node} node The element.
261    * @private
262    */
263   function initAnimation_(node) {
264     /** @type {number} */
265     var width = parseFloat(node.style.width) || node.offsetWidth;
266     /** @type {number} */
267     var height = parseFloat(node.style.height) || node.offsetHeight;
268     /** @type {number} */
269     var x = parseFloat(node.style.left) || node.offsetLeft;
270     /** @type {number} */
271     var y = parseFloat(node.style.top) || node.offsetTop;
272     /** @type {number} */ var direction = getDirection_();
273 
274     if (direction == charts.Grid.DIRECTION.LEFT_TO_RIGHT) {
275       node.style.width = '1px';
276       animation.animate(node, {'width': width});
277     } else if (direction == charts.Grid.DIRECTION.RIGHT_TO_LEFT) {
278       node.style.width = '1px';
279       node.style.right = '1px';
280       node.style.left = 'auto';
281       animation.animate(node, {'width': width});
282     } else if (direction == charts.Grid.DIRECTION.TOP_TO_BOTTOM) {
283       node.style.height = '1px';
284       animation.animate(node, {'height': height});
285     } else if (direction == charts.Grid.DIRECTION.BOTTOM_TO_TOP) {
286       node.style.height = '1px';
287       node.style.bottom = '1px';
288       node.style.top = 'auto';
289       animation.animate(node, {'height': height});
290     }
291   }

293   /**
294    * @param {!Element|Node} node The element.
295    * @return {!Object.<string, number>} Returns tooltip position as {x, y}.
296    * @private
297    */
298   function getTooltipPosition_(node) {
299     /** @type {number} */ var width = node.offsetWidth;
300     /** @type {number} */ var height = node.offsetHeight;
301     /** @type {!Object.<string, number>} */
302     var result = {x: node.offsetLeft, y: node.offsetTop};
303 
304     while (node = node.offsetParent) {
305       result.x += node.offsetLeft;
306       result.y += node.offsetTop;
307     }
308 
309     /** @type {Node} */ var tooltip = self_.tooltip.getTooltipElement();
310     /** @type {number} */ var direction = getDirection_();
311     if (direction == charts.Grid.DIRECTION.LEFT_TO_RIGHT) {
312       result.x += width;
313       result.y += options_['stroke'];
314     } else if (direction == charts.Grid.DIRECTION.RIGHT_TO_LEFT) {
315       result.x -= tooltip.offsetWidth;
316       result.y += options_['stroke'];
317     } else if (direction == charts.Grid.DIRECTION.BOTTOM_TO_TOP) {
318       result.x += options_['stroke'];
319       result.y -= tooltip.offsetHeight;
320     } else if (direction == charts.Grid.DIRECTION.TOP_TO_BOTTOM) {
321       result.x += options_['stroke'];
322       result.y += height;
323     }
324     return result;
325   }
