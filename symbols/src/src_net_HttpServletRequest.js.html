 10 /**
 11  * Simple implementation of javax.servlet.http.HttpServletRequest.
 12  * @constructor
 13  * @extends {net.ServletRequest} net.ServletRequest
 14  * @requires dom.scripts
 15  */
 16 net.HttpServletRequest = function() {
 17   net.ServletRequest.apply(this, arguments);
 18 
 19   /**
 20    * Returns the query string that is contained in the request URL after the
 21    *     path.
 22    * @return {string} Returns the query string that is contained in the request
 23    *     URL after the path.
 24    */
 25   this.getQueryString = function() {
 26     return location.search && location.search.substr(1);
 27   };
 28 
 29   /**
 30    * Returns the portion of the request URI that indicates the context of the
 31    * request.
 32    * @return {string} Returns current context path.
 33    */
 34   this.getContextPath = function() {
 35     if (!net.HttpServletRequest.contextPath_) {
 36       /** @type {Element} */ var script = dom.scripts.getCurrent();
 37       if (script) {
 38         /** @type {string} */ var path = script.getAttribute('src');
 39         /** @type {Element} */ var a = dom.createElement('A');
 40         a['href'] = path;
 41 
 42         // Replace all relatives.
 43         path = path.replace(/(\.+\/)+/, '');
 44         net.HttpServletRequest.contextPath_ = a['pathname'].replace(path, '');
 45       }
 46     }
 47     return net.HttpServletRequest.contextPath_ || '';
 48   };
 49 };
