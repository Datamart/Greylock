 10 /**
 11  * Constructor of InputNumber.
 12  * An implementation of HTML5 input type="number" for non HTML5 browsers.
 13  * @param {string|Element} input The input element or its ID attribute.
 14  * @constructor
 15  * @link http://www.w3.org/TR/html-markup/input.number.html
 16  */
 17 controls.html5.InputNumber = function(input) {
 18 
 19   /**
 20    * @private
 21    */
 22   function init_() {
 23     // @bug: https://bugzilla.mozilla.org/show_bug.cgi?id=344616
 24     if ('number' != input_.type) {
 25       dom.events.addEventListener(input_, dom.events.TYPE.KEYDOWN, keydown_);
 26       dom.events.addEventListener(
 27           input_, dom.events.TYPE.MOUSEDOWN, mousedown_);
 28       dom.events.addEventListener(
 29           input_, dom.events.TYPE.MOUSEMOVE, mousemove_);
 30       if (!input_.getAttribute('pattern')) {
 31         input_.setAttribute('pattern', '^\-?[0-9]+$');
 32       }
 33       if (!input_.getAttribute('maxlength')) {
 34         input_.setAttribute('maxlength', ('' + max_).length);
 35       }
 36       input_.style.backgroundImage = 'url(data:image/png;base64,' + img_ + ')';
 37       input_.style.backgroundPosition = 'right center';
 38       input_.style.backgroundRepeat = 'no-repeat';
 39       input_.style.paddingRight = padding_ + 'px';
 40     }
 41   }
 42 
 43   /**
 44    * @private
 45    */
 46   function increment_() {
 47     /**@type {number} */ var value = parseInt(input_.value, 10) || 0;
 48     input_.value = Math.min(value + step_, max_);
 49     dispatchEvents_();
 50   }
 51 
 52   /**
 53    * @private
 54    */
 55   function decrement_() {
 56     /**@type {number} */ var value = parseInt(input_.value, 10) || 0;
 57     input_.value = Math.max(value - step_, min_);
 58     dispatchEvents_();
 59   }
 60 
 61   /**
 62    * @param {Event} e The keydown event.
 63    * @private
 64    */
 65   function keydown_(e) {
 66     e = e || window.event;
 67     if (isNaN(input_.value)) {
 68       input_.focus();
 69     } else {
 70       if (e.keyCode == 38) increment_();
 71       else if (e.keyCode == 40) decrement_();
 72     }
 73   }
 74 
 75   /**
 76    * @param {Event} e The mousedown event.
 77    * @private
 78    */
 79   function mousedown_(e) {
 80     e = e || window.event;
 81     /** @type {Object} */ var rect = dom.getBoundingClientRect(input_);
 82     /** @type {number} */ var x = e.clientX;
 83     /** @type {number} */ var y = e.clientY;
 84     if (x && x > rect['right'] - padding_) {
 85       // @bug MSIE does not calculates rect.height for inline elements.
 86       /** @type {number} */ var height = rect['height'] || input_.offsetHeight;
 87       if (y < rect['top'] + height / 2) {
 88         increment_();
 89       } else {
 90         decrement_();
 91       }
 92     }
 93   }
 94 
 95   /**
 96    * @param {Event} e The mousemove event.
 97    * @private
 98    */
 99   function mousemove_(e) {
100     e = e || window.event;
101     /** @type {Object} */ var rect = dom.getBoundingClientRect(input_);
102     /** @type {number} */ var x = e.clientX;
103     if (x && x > rect['right'] - padding_) {
104       input_.style.cursor = 'default';
105     } else {
106       input_.style.cursor = 'text';
107     }
108   }
109 
110   /**
111    * Dispatches input events.
112    * @return {boolean} Returns <code>true</code> if all events are
113    * dispatched successfully.
114    * @private
115    */
116   function dispatchEvents_() {
117     /** @type {boolean} */
118     var result = dom.events.dispatchEvent(input_, 'input');
119     result = dom.events.dispatchEvent(input_, dom.events.TYPE.CHANGE) && result;
120     return result;
121   }
122 
123   /**
124    * The reference to input element.
125    * @type {Element}
126    * @private
127    */
128   var input_ = typeof input == 'string' ? dom.getElementById(input) : input;
129 
130   /**
131    * The expected lower bound for the input value.
132    * @type {number}
133    * @private
134    */
135   var min_ = +(input_.getAttribute('min') || -1e6);
136 
137   /**
138    * The expected upper bound for the input value.
139    * @type {number}
140    * @private
141    */
142   var max_ = +(input_.getAttribute('max') || 1e6);
143 
144   /**
145    * Specifies the value granularity of the input value.
146    * @type {number}
147    * @private
148    */
149   var step_ = +(input_.getAttribute('step') || 1);
150 
151   /**
152    * Base64 encoded PNG top-bottom arrows image.
153    * @type {string}
154    * @private
155    */
156   var img_ = 'iVBORw0KGgoAAAANSUhEUgAAAA8AAAAQBAMAAAA7eDg3AAAAA3NCSVQICAjb4U' +
157              '/gAAAACXBIWXMAAAsSAAALEgHS3X78AAAAHHRFWHRTb2Z0d2FyZQBBZG9iZSBG' +
158              'aXJld29ya3MgQ1M26LyyjAAAABR0RVh0Q3JlYXRpb24gVGltZQA2LzYvMTOlfn' +
159              '1pAAAAElBMVEUAAABmZmb///9kZGRmZmZmZmaw6hPAAAAABXRSTlMAAAAB/Rh8' +
160              'uVEAAAArSURBVAjXY1CCAAUGvAwFViYoIzQIwlANDYWKMDChMRBSroEw7UYErY' +
161              'AwAOxWEY6fpOynAAAAAElFTkSuQmCC';
162 
163   /**
164    * Input's right padding. (Actual background image width.)
165    * @type {number}
166    * @private
167    */
168   var padding_ = 16;
169 
170   init_();
171 };
