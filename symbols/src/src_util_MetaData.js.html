 10 /**
 11  * Constructor of MetaData.
 12  * @class Constructor of MetaData.
 13  * @extends {util.PropertyReader} util.PropertyReader
 14  * @constructor
 15  * @example
 16  * <b>var</b> meta = <b>new</b> util.MetaData();
 17  * console.log(meta.getValues());
 18  * console.log(meta.getValues(document));
 19  * console.log(meta.getValues(document.getElementById('container')));
 20  * console.log(meta.getValues(ajax.responseXML));
 21  */
 22 util.MetaData = function() {
 23   util.PropertyReader.apply(this, arguments);
 24 
 25   /**
 26    * @param {Document|Node=} opt_context The optional DOM context.
 27    * @return {Object.<string, string>} Returns meta data.
 28    */
 29   this.getValues = function(opt_context) {
 30     self_.read(opt_context || dom.document, 'meta', 'name', 'content');
 31     return self_.getData();
 32   };
 33 
 34   /**
 35    * @type {!util.MetaData}
 36    * @private
 37    */
 38   var self_ = this;
 39 };
