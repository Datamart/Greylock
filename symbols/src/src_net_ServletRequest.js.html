 11 /**
 12  * Simple implementation of javax.servlet.ServletRequest.
 13  * @requires util.StringUtils.URI
 14  * @constructor
 15  * @see http://docs.oracle.com/javaee/5/api/javax/servlet/ServletRequest.html
 16  */
 17 net.ServletRequest = function() {
 18 
 19   /**
 20    * Returns the value of a request parameter as a string, or empty string if
 21    * the parameter does not exist.
 22    * @param {string} name A <code>string</code> specifying the name of the
 23    *     parameter.
 24    * @param {Element|Location|string=} opt_location Optional location.
 25    * @return {string} Returns a <code>string</code> representing the single
 26    *     value of the parameter.
 27    * @this {net.ServletRequest}
 28    */
 29   this.getParameter = function(name, opt_location) {
 30     /** @type {!Object.<string, string>} */
 31     var map = this.getParameterMap(opt_location);
 32     return map[/** @type {string} */ (name)] || '';
 33   };
 34 
 35   /**
 36    * Returns a map of the parameters of this request including parameters from
 37    * parsed from query string and hash.
 38    * @param {Element|Location|string=} opt_location Optional location.
 39    * @return {!Object.<string, string>} Map containing parameter names as keys
 40    *     and parameter values as map values.
 41    */
 42   this.getParameterMap = function(opt_location) {
 43     /** @type {!Object.<string, string>} */ var map = {};
 44     /** @type {!Array.<string>} */ var pairs = parseLocation_(opt_location);
 45     /** @type {number} */ var index = pairs.length >>> 0;
 46     while (index--) {
 47       /** @type {!Array.<string>} */ var pair = pairs[index].split('=');
 48       /** @type {string} */ var key = pair[0];
 49       if (key) map[key] = util.StringUtils.URI.decode(pair[1]);
 50     }
 51     return map;
 52   };
 53 
 54   /**
 55    * Gets list of parameters pairs parsed form <code>opt_location</code>.
 56    * @param {Element|Location|string=} opt_location Optional location.
 57    * @return {!Array.<string>} Returns list of parameters pairs.
 58    * @private
 59    */
 60   function parseLocation_(opt_location) {
 61     opt_location = opt_location || location;
 62 
 63     /** @type {string} */ var hash = 'string' == typeof opt_location ?
 64         (opt_location.split('#')[1] || '') :
 65         opt_location.hash.substr(1);
 66 
 67     /** @type {string} */ var query = 'string' == typeof opt_location ?
 68         (opt_location.split('?')[1] || '').split('#')[0] :
 69         opt_location.search.substr(1);
 70 
 71     return query.split('&').concat(hash.split('&'));
 72   }
 73 };
