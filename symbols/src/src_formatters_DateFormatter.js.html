 10 /**
 11  * Constructor of DateFormatter.
 12  * @requires locale.Calendar
 13  * @constructor
 14  */
 15 formatters.DateFormatter = function() {
 16 
 17   /**
 18    * Formats given <code>date</code> according to given <code>format</code>.
 19    * @param {Date} date The Date to be formatted.
 20    * @param {string} format The date format.
 21    * @return {string} The formatted date as string.
 22    * @example
 23    * <b>var</b> formatter = <b>new</b> formatters.DateFormatter();
 24    * formatter.format(<b>new</b> Date(), 'YYYY-MM-dd');
 25    * formatter.format(<b>new</b> Date(), 'YYYY-MM-dd hh:mm');
 26    * formatter.format(<b>new</b> Date(), 'dd MMM, YYYY');
 27    */
 28   this.format = function(date, format) {
 29     /** @type {string} */ var month = calendar_.getMonthName(date);
 30     /** @type {!Array.<string, number>} */ var tokens = [
 31       'YYYY', date.getFullYear(),
 32       'YY', (date.getFullYear() + '').substr(2),
 33       'MMMM', month,
 34       'MMM', month.substr(0, 3),
 35       'MM', ('0' + (date.getMonth() + 1)).slice(-2),
 36       'dd', ('0' + date.getDate()).slice(-2),
 37       'hh', ('0' + date.getHours()).slice(-2),
 38       'mm', ('0' + date.getMinutes()).slice(-2),
 39       'ss', ('0' + date.getSeconds()).slice(-2)
 40     ];
 41     /** @type {number} */ var length = tokens.length;
 42     for (/** @type {number} */ var i = 0; i < length;) {
 43       format = format.replace(tokens[i++], tokens[i++]);
 44     }
 45     return format;
 46   };
 47 
 48   /**
 49    * Parses given date <code>str</code>  according to given <code>format</code>.
 50    * @param {string} str The date string to be parsed.
 51    * @param {string} format The date format.
 52    * @return {Date} The parsed date.
 53    * @example
 54    * <b>var</b> formatter = <b>new</b> formatters.DateFormatter();
 55    * formatter.parse('2013-01-31', 'YYYY-MM-dd');
 56    * formatter.parse('2013-01-31 20:30', 'YYYY-MM-dd hh:mm');
 57    * formatter.parse('03 Jan, 2013', 'dd MMM, YYYY');
 58    */
 59   this.parse = function(str, format) {
 60     /** @type {!RegExp} */ var re = /[\/\s,\.\:\-]/;
 61     /** @type {!Array.<string>} */ var dateParts = str.split(re);
 62     /** @type {!Array.<string>} */ var formatParts = format.split(re);
 63     /** @type {!Object.<string, string>} */ var map = {};
 64     /** @type {number} */ var i;
 65     for (i = 0; i < Math.min(dateParts.length, formatParts.length); i++) {
 66       /** @type {string} */ var key = formatParts[i];
 67       if (key) map[key] = dateParts[i];
 68     }
 69 
 70     /** @type {number} */ var year = +(map['YYYY'] || '20' + map['YY']);
 71     /** @type {number} */ var month = +(map['MM'] || 0) - 1;
 72     if (map['MMMM'] || map['MMM']) {
 73       month = calendar_.getMonthByName(map['MMMM'] || map['MMM']);
 74     }
 75     return month >= 0 && +map['dd'] < 32 ?
 76         new Date(year, month, +map['dd'], +map['hh'] || 0,
 77                +map['mm'] || 0, +map['ss'] || 0) : dom.NULL;
 78   };
 79 
 80   /**
 81    * @type {!locale.Calendar}
 82    * @private
 83    */
 84   var calendar_ = new locale.Calendar;
 85 };

 88 /**
 89  * Formats given <code>date</code> according to given <code>format</code>.
 90  * @param {Date} date The Date to be formatted.
 91  * @param {string} format The date format.
 92  * @return {string} The formatted date as string.
 93  * @static
 94  * @example
 95  * formatters.DateFormatter.format(<b>new</b> Date(), 'YYYY-MM-dd');
 96  * formatters.DateFormatter.format(<b>new</b> Date(), 'YYYY-MM-dd hh:mm');
 97  * formatters.DateFormatter.format(<b>new</b> Date(), 'dd MMM, YYYY');
 98  */
 99 formatters.DateFormatter.format = function(date, format) {
100   if (!formatters.DateFormatter.formatter_) {
101     formatters.DateFormatter.formatter_ = new formatters.DateFormatter;
102   }
103   return formatters.DateFormatter.formatter_.format(date, format);
104 };

109  * @param {string} str The date string to be parsed.
110  * @param {string} format The date format.
111  * @return {Date} The parsed date.
112  * @static

117  * formatter.parse('03 Jan, 2013', 'dd MMM, YYYY');

121     formatters.DateFormatter.formatter_ = new formatters.DateFormatter;
