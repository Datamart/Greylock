 10 /**
 11  * The constructor initiates a XMLHttpRequest.
 12  * @constructor
 13  */
 14 net.HttpRequest = function() {
 15 
 16   /**
 17    * Performs GET request.
 18    * @param {string} url The URL to which to send the request.
 19    * @param {function(XMLHttpRequest)} callback A JavaScript function object
 20    *     that is called whenever the <code>readyState</code> attribute equals
 21    *     to 4 (DONE).
 22    * @return {!XMLHttpRequest} Returns instance of XMLHttpRequest.
 23    */
 24   this.doGet = function(url, callback) {
 25     // @link http://tools.ietf.org/html/rfc2616#section-3.2.1
 26     if (url.length > 255) {
 27       /** @type {!Array} */ var parts = url.split('?');
 28       return self_.doPost(parts[0], callback, parts[1]);
 29     }
 30     return doRequest_('GET', url, callback);
 31   };
 32 
 33   /**
 34    * Performs POST request.
 35    * @param {string} url The URL to which to send the request.
 36    * @param {function(XMLHttpRequest)} callback A JavaScript function object
 37    *     that is called whenever the <code>readyState</code> attribute equals
 38    *     to 4 (DONE).
 39    * @param {Object|string} data Provides the request entity body.
 40    * @return {!XMLHttpRequest} Returns instance of XMLHttpRequest.
 41    */
 42   this.doPost = function(url, callback, data) {
 43     return doRequest_('POST', url, callback, data);
 44   };
 45 
 46   /**
 47    * Performs HEAD request.
 48    * @param {string} url The URL to which to send the request.
 49    * @param {function(XMLHttpRequest)} callback A JavaScript function object
 50    *     that is called whenever the <code>readyState</code> attribute equals
 51    *     to 4 (DONE).
 52    * @return {!XMLHttpRequest} Returns instance of XMLHttpRequest.
 53    */
 54   this.doHead = function(url, callback) {
 55     return doRequest_('HEAD', url, callback);
 56   };
 57 
 58   /**
 59    * Gets count of active requests.
 60    * @return {number} Returns count of active requests.
 61    */
 62   this.getCount = function() {
 63     return net.HttpRequest.count_ || 0;
 64   };
 65 
 66   /**
 67    * Performs HTTP request.
 68    * @param {string} method The HTTP method to use, such as "GET", "POST",
 69    *     "PUT", "DELETE", etc. Ignored for non-HTTP(S) URLs.
 70    * @param {string} url The URL to which to send the request.
 71    * @param {function(XMLHttpRequest)} callback A JavaScript function object
 72    *     that is called whenever the <code>readyState</code> attribute equals
 73    *     to 4 (DONE).
 74    * @param {string|Object=} opt_data Provides the request entity body.
 75    *     If opt_data is Object, request will be sent as application/json.
 76    * @return {!XMLHttpRequest} Returns instance of XMLHttpRequest.
 77    * @private
 78    */
 79   function doRequest_(method, url, callback, opt_data) {
 80     url = url.replace(/^\/+/, '/');
 81     //url += (url.indexOf('?') + 1 ? '&' : '?') + 'nocache=' + (+new Date);
 82     url += (~url.indexOf('?') ? '&' : '?') + 'nocache=' + (+new Date);
 83     net.HttpRequest.count_++;
 84 
 85     /** @type {!XMLHttpRequest} */
 86     var req = window.XMLHttpRequest ? new XMLHttpRequest :
 87                                       new ActiveXObject('Microsoft.XMLHTTP');
 88     req.onreadystatechange = function() {
 89       if (req.readyState == 4) {
 90         net.HttpRequest.count_--;
 91         callback(req);
 92       }
 93     };
 94     req.open(method, url, true);
 95     req.send(prepare_(req, method, opt_data));
 96     return req;
 97   }
 98 
 99   /**
100    * Sets request <code>request</code> headers depending to request
101    * <code>method</code> and <code>opt_data</code>.
102    * @param {!XMLHttpRequest} request The instance of XMLHttpRequest.
103    * @param {string} method The HTTP method to use, such as "GET", "POST",
104    *     "PUT", "DELETE", etc. Ignored for non-HTTP(S) URLs.
105    * @param {string|Object=} opt_data Provides the request entity body.
106    *     If opt_data is Object, request will be sent as application/json.
107    * @return {string} Returns modified <code>opt_data</code>.
108    * @private
109    */
110   function prepare_(request, method, opt_data) {
111     for (var name in net.HttpRequest.headers_) {
112       request.setRequestHeader(name, net.HttpRequest.headers_[name]);
113     }
114 
115     request.setRequestHeader('X-Requested-With', 'XMLHttpRequest');
116 
117     if (opt_data && typeof opt_data == 'object') {
118       request.setRequestHeader('Content-Type',
119           'application/json; charset=' + dom.CHARSET);
120       opt_data = util.StringUtils.JSON.stringify(
121           /** @type {!Object} */ (opt_data));
122     } else if (method == 'POST') {
123       request.setRequestHeader('Content-Type',
124           'application/x-www-form-urlencoded; charset=' + dom.CHARSET);
125     }
126 
127     return /** @type {string} */ (opt_data);
128   }
129 
130   /**
131    * The reference to current class instance.
132    * Used in private methods and for preventing jslint errors.
133    * @type {!net.HttpRequest}
134    * @private
135    */
136   var self_ = this;
137 
138   /**
139    * The count of active requests.
140    * @static
141    * @type {number}
142    * @see net.HttpRequest#getCount
143    * @private
144    */
145   net.HttpRequest.count_ = net.HttpRequest.count_ || 0;
146 };
147

150  * @static

152  * @see net.HttpRequest.setRequestHeader
153  * @private
154  */
155 net.HttpRequest.headers_ = {};

158 /**
159  * Sets custom header.
160  * @param {string} name The header name.
161  * @param {string} value The header value.
162  * @static
163  */
164 net.HttpRequest.setRequestHeader = function(name, value) {
165   net.HttpRequest.headers_[name] = value;
166 };
