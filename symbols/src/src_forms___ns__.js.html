 24 /**
 25  * Initializes HTML5 form features: placeholder, validation, etc.
 26  * @param {Node} container The HTML container which contains form elements.
 27  * @static
 28  * @example
 29  * forms.init(document.forms[0]);
 30  */
 31 forms.init = function(container) {
 32   (new forms.PlaceHolder).init(container);
 33   (new forms.Validation).init(container);
 34 };

 37 /**
 38  * Adds onchange/oninput event handler on all form elements.
 39  * @param {HTMLFormElement} form Form element.
 40  * @param {!Function} handler Event handler.
 41  */
 42 forms.onchange = function(form, handler) {
 43 
 44   function checker_(element, handler) {
 45     /** @type {string} */ var value = element.value;
 46     setInterval(function() {
 47       if (value != element.value) {
 48         handler(/** @type {Event} */({'target': element}));
 49         value = element.value;
 50       }
 51     }, 99);
 52   }
 53 
 54   if (form) {
 55     /** @type {HTMLCollection} */ var elements = form.elements;
 56     /** @type {number} */ var length = elements.length;
 57     while (length) checker_(elements[--length], handler);
 58   }
 59 };
