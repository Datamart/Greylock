 11 /**
 12  * GeoChart constructor.
 13  * @param {string|Element} container The HTML container.
 14  * @constructor
 15  * @extends {charts.BaseChart} charts.BaseChart
 16  * @requires charts.GeoLocations
 17  * @requires formatters.NumberFormatter
 18  * @example
 19  * <b>var</b> chart = <b>new</b> charts.GeoChart('container_id');
 20  * chart.draw([['Country', 'Population'],
 21  *             ['Germany', 200],
 22  *             ['USA',     300],
 23  *             ['Brazil',  400],
 24  *             ['Canada',  500],
 25  *             ['France',  600],
 26  *             ['Russia',  700]
 27  * ]);
 28  *
 29  * <div style="border: solid 1px #ccc; margin: 5px; padding: 5px; width: 560px">
 30  *   <div id="chart-container"
 31  *        style="width: 560px; height: 365px;"></div>
 32  * </div>
 33  * <script src="../../bin/jscb.js"></script>
 34  * <script>
 35  *   var chart = new charts.GeoChart('chart-container');
 36  *   chart.draw([['Country', 'Population'],
 37  *               ['Germany', 200],
 38  *               ['USA',     300],
 39  *               ['Brazil',  400],
 40  *               ['Canada',  500],
 41  *               ['France',  600],
 42  *               ['Russia',  700]
 43  *   ]);
 44  * </script>
 45  */
 46 charts.GeoChart = function(container) {
 47   charts.BaseChart.apply(this, arguments);
 48 
 49   /**
 50    * Country name mapping. (e.g. usa: United States).
 51    * @enum {string}
 52    */
 53   var COUNTRY_MAPPING = {
 54     'usa': 'United States',
 55     'russia': 'Russian Federation',
 56     'drc': 'Democratic Republic of the Congo'
 57   };
 58 
 59   /**
 60    * Original svg viewport width: 950px height: 620px.
 61    * Used for scaling map to different sizes.
 62    * @enum {number}
 63    */
 64   var VIEWPORT = {
 65     width: 950,
 66     height: 620
 67   };
 68 
 69   /**
 70    * Draws the chart based on <code>data</code> and <code>opt_options</code>.
 71    * @param {!Array.<Array>} data A chart data.
 72    * @param {Object=} opt_options A optional chart's configuration options.
 73    * @override
 74    */
 75   this.draw = function(data, opt_options) {
 76     options_ = getOptions_(opt_options);
 77     self_.tooltip.setOptions(options_);
 78     /** @type {number} */ var width = self_.container.offsetWidth || 200;
 79     /** @type {number} */ var height = self_.container.offsetHeight || width;
 80 
 81     /** @type {string} */ var content = '';
 82     /** @type {number} */
 83     var scale = Math.min(width / VIEWPORT.width, height / VIEWPORT.height);
 84 
 85     /** @type {!Object.<string, Object>} */ var map = getColorMap_(data);
 86     /** @type {!Array.<Array>} */ var locations = charts.GeoLocations.DATA;
 87     for (/** @type {number} */ var i = 0; i < locations.length; i++) {
 88       /** @type {string} */ var country = getCountryKey_(locations[i][0]);
 89       /** @type {string} */ var path = locations[i][1];
 90       /** @type {number} */ var value = country in map ? map[country].value : 0;
 91       /** @type {string} */ var color = country in map ? map[country].color :
 92           /** @type {string} */ (options_['fillcolor']);
 93       /** @type {string} */
 94       var column = country in map ? map[country].column : '';
 95       /** @type {string} */
 96       var tooltip = '<b>' + getCountryName_(country) + '</b>';
 97       if (value) {
 98         tooltip += '<br>' + column + ': ';
 99         tooltip += 'number' == typeof value ? formatter_.format(value) : value;
100       }
101       content += (charts.IS_SVG_SUPPORTED ? getSvgContent_ : getVmlContent_)(
102           path, tooltip, color, scale, country);
103     }
104 
105     self_.drawContent(content, width, height);
106     initEvents_();
107   };
108 
109   // Export for closure compiler.
110   this['draw'] = this.draw;
111 
112   /**
113    * @param {Array} data The data.
114    * @return {!Object.<string, Object>} Returns data colors.
115    * @private
116    */
117   function getColorMap_(data) {
118     /** @type {!Object.<string, Object>} */ var colors = {};
119     /** @type {Array.<number>} */ var range = self_.getDataRange(data, 1);
120     /** @type {!Array.<Array>} */ var rows = self_.getDataRows(data);
121     /** @type {number} */ var min = range[0];
122     /** @type {number} */ var max = range[1];
123     /** @type {string} */ var column = data[0][1];
124     for (/** @type {number} */ var i = 0; i < rows.length; i++) {
125       /** @type {string} */
126       var country = getCountryKey_(rows[i][0]).toLowerCase();
127       /** @type {number} */ var value = rows[i][1];
128       /** @type {number} */
129       var weight = (+value - min) / (max - min) * 0.8 + 0.2;
130       weight = weight ? weight : 1 * 0.8;
131       /** @type {string} */ var color = 'rgb(' +
132           getColor_(options_['rgb']['red'], weight) + ',' +
133           getColor_(options_['rgb']['green'], weight) + ',' +
134           getColor_(options_['rgb']['blue'], weight) + ')';
135       colors[country] = {color: color, value: value, column: column};
136     }
137     return colors;
138   }

140   /**
141    * @param {number} component Color component: red, green or blue.
142    * @param {number} weight Color weight.
143    * @return {number} Returns color component.
144    * @private
145    */
146   function getColor_(component, weight) {
147     return parseInt(255 - (255 - component) * weight, 10);
148   }

150   /**
151    * @param {string} country The country name.
152    * @return {string} Returns country key.
153    * @private
154    */
155   function getCountryKey_(country) {
156     for (var key in COUNTRY_MAPPING)
157       if (COUNTRY_MAPPING[key].toLowerCase() == country.toLowerCase())
158         return key;
159       return country;
160   }
161 
162   /**
163    * @param {string} country The country key.
164    * @return {string} Returns country name.
165    * @private
166    */
167   function getCountryName_(country) {
168     // Exceptional cases:
169     if (country in COUNTRY_MAPPING)
170       return COUNTRY_MAPPING[country];
171 
172     /** @type {Array.<string>} */
173     var words = country.replace(/\s+/g, '_').split('_');
174     for (var i = 0; i < words.length; i++) {
175       words[i] = words[i].charAt(0).toUpperCase() + words[i].substr(1);
176     }
177     return words.join(' ');
178   }

180   /**
181    * @param {string} path The path.
182    * @param {string} tooltip The tooltip.
183    * @param {string} color The color.
184    * @param {number} scale The scale.
185    * @param {string} country The country key.
186    * @return {string} Returns SVG markup string.
187    * @private
188    */
189   function getSvgContent_(path, tooltip, color, scale, country) {
190     return '<path transform="scale(' + scale + ')" ' +
191            'tooltip="' + tooltip + '" d="' + path + '"' +
192            'fill="' + color + '" ' +
193            'id="' + country + '" ' +
194            'stroke="' + options_['stroke']['color'] + '"/>';
195   }

197   /**
198    * @param {string} path The path.
199    * @param {string} tooltip The tooltip.
200    * @param {string} color The color.
201    * @param {number} scale The scale.
202    * @param {string} country The country key.
203    * @return {string} Returns VML markup string.
204    * @private
205    */
206   function getVmlContent_(path, tooltip, color, scale, country) {
207     //path = getVmlPath_(path);
208     path = graphics.VmlHelper.getVmlPath(path);
209 
210     /** @type {number} */ var width = self_.container.offsetWidth || 200;
211     /** @type {number} */ var height = self_.container.offsetHeight || width;
212 
213     /** @type {Array.<number>} */
214     var coordsize = [width / scale, height / scale];
215 
216     /** @type {string} */
217     var vml = '<v:shape fillcolor="' + color + '" ' +
218         'strokecolor="' + options_['stroke']['color'] + '" ' +
219         'coordorigin="0 0" ' +
220         'coordsize="' + coordsize + '" ' +
221         'style="width:100%;height:100%" ' +
222         'id="' + country + '" ' +
223         'tooltip="' + tooltip + '">' +
224         '<v:path v="' + path + '"/>' +
225         '</v:shape>';
226     return vml;
227   }

229   /**
230    * Converts SVG path to VML path.
231    * @param {string} svgPath The SVG path.
232    * @return {string} Returns VML path.
233    * @deprecated Use graphics.VmlHelper.getVmlPath instead.
234    * @private
235    */
236   function getVmlPath_(svgPath) {
237     // TODO (alex): Calculate VML path.
238     // M387.56,224.4l-0.54,1.37l0.81,0.82l2.17-1.37L387.56,224.4L387.56,224.4z
239     svgPath = svgPath.replace(/(\d*)((\.*\d*)(e ?-?\d*))/g, '$1');
240     /** @type {Array.<string>} */
241     var commands = svgPath.match(/([MLHVCSQTAZ].*?)(?=[MLHVCSQTAZ]|$)/gi);
242     /** @type {string} */ var vmlPath = '';
243     /** @type {number} */ var cursorX = 0;
244     /** @type {number} */ var cursorY = 0;
245     for (/** @type {number} */ var i = 0; i < commands.length; i++) {
246       /** @type {string} */ var command = commands[i].substring(0, 1);
247       /** @type {!Array} */
248       var params = commands[i].substring(1, commands[i].length).split(/[, ]/);
249       for (/** @type {number} */ var j = 0; j < params.length; j++) {
250         //params[j] = Math.round(parseFloat(params[j]));
251       }
252       /** @type {string} */ var args = params.join();
253       /** @type {!Array} */ var coords = args.split(/[, ]+/);
254       switch (command) {
255         case 'M': // moveTo absolute
256           command = 'm';
257           cursorX = parseInt(coords[0], 10);
258           cursorY = parseInt(coords[1], 10);
259           break;
260         case 'm': // moveTo relative
261           command = 't';
262           coords[0] = parseInt(coords[0], 10) + parseInt(cursorX, 10);
263           coords[1] = parseInt(coords[1], 10) + parseInt(cursorY, 10);
264           cursorX = parseInt(coords[0], 10);
265           cursorY = parseInt(coords[1], 10);
266           //args = coords[0] + ',' + coords[1] + ' ';
267           break;
268         case 'A': // arc absolute:
269           // SVG: rx ry x-axis-rotation large-arc-flag sweep-flag x y
270           // VML: center (x,y) size(w,h) start-angle, end-angle
271           command = 'ae';
272           coords[0] = parseInt(coords[0], 10);
273           coords[1] = parseInt(coords[1], 10);
274 
275           coords[2] = parseInt(coords[2], 10);
276           coords[3] = parseInt(coords[3], 10);
277 
278           coords[4] = parseInt(coords[4], 10);
279           coords[5] = parseInt(coords[5], 10);
280           args = coords[4] + ' ' + coords[5] + ' ' + (coords[2] * 2) + ' ' +
281                  (coords[3] * 2) + ' 0 360';
282           break;
283         case 'L': // lineto absolute
284         case 'H': // horizontal lineto absolute
285           command = 'l';
286           cursorX = parseInt(coords[0], 10);
287           cursorY = parseInt(coords[1], 10);
288           break;
289         case 'l': // lineto relative
290           command = 'r';
291           //window.console && console.log(coords);
292           coords[0] = parseInt(coords[0], 10) + parseInt(cursorX, 10);
293           coords[1] = parseInt(coords[1], 10) + parseInt(cursorY, 10);
294           cursorX = parseInt(coords[0], 10);
295           cursorY = parseInt(coords[1], 10);
296           // args = coords[0] + ',' + coords[1] + ' ';
297           break;
298         case 'h': // horizontal lineto relative
299           command = 'r';
300           //window.console && console.log(coords);
301           coords[0] = parseInt(coords[0], 10) + parseInt(cursorX, 10);
302           coords[1] = cursorY;
303           cursorX = parseInt(coords[0], 10);
304           cursorY = parseInt(coords[1], 10);
305           args = coords[0] + ',' + coords[1] + ' ';
306           break;
307         case 'c':
308           command = 'v';
309           break;
310         case 'z':
311           command = 'xe';
312           args = '';
313         default:
314           command = command.toLowerCase();
315       }
316       vmlPath += command + args;
317     }
318     return vmlPath;
319   }

321   /**
322    * Initializes events handlers.
323    * @private
324    */
325   function initEvents_() {
326     /** @type {Array.<Element>} */ var paths = null;
327     /** @type {!Object.<string, number>} */ var tags = {'path': 0, 'shape': 0};
328 
329     dom.events.addEventListener(
330         self_.container, dom.events.TYPE.MOUSEOUT, function(e) {
331           if (paths) {
332             for (/** @type {number} */ var i = 0; i < paths.length; i++) {
333               if (charts.IS_SVG_SUPPORTED) {
334                 //paths[i].setAttribute('stroke-width',
335                 //                  options_['stroke']['width'] + 'px');
336                 paths[i].setAttribute('stroke',
337                                   options_['stroke']['color']);
338               } else {
339                 //paths[i]['strokeweight'] = options_['stroke']['width'] + 'px';
340                 paths[i]['strokecolor'] = options_['stroke']['color'];
341               }
342             }
343             self_.tooltip.hide(e);
344             paths = null;
345           }
346         });
347 
348     dom.events.addEventListener(
349         self_.container, dom.events.TYPE.MOUSEMOVE, function(e) {
350           dom.events.dispatchEvent(self_.container, dom.events.TYPE.MOUSEOUT);
351           e = e || window.event;
352           /** @type {Element} */
353           var target = e.target || e.srcElement || e.toElement;
354           if (target && target.tagName in tags) {
355             /** @type {NodeList} */
356             var nodes = dom.getElementsByTagName(target.parentNode,
357                                                  target.tagName);
358             paths = [];
359             for (/** @type {number} */ var i = 0; i < nodes.length; i++) {
360               /** @type {Element} */ var node = nodes[i];
361               if (node.id == target.id) {
362                 paths.push(node);
363               }
364             }
365             for (i = 0; i < paths.length; i++) {
366               if (charts.IS_SVG_SUPPORTED) {
367                 //paths[i].setAttribute('stroke-width', '3px');
368                 paths[i].setAttribute('stroke', '#666');
369               } else {
370                 //paths[i]['strokeweight'] = '3px';
371                 paths[i]['strokecolor'] = '#666';
372               }
373             }
374             self_.tooltip.show(e);
375           }
376         });
377   }

379   /**
380    * Gets chart's options merged with defaults chart's options.
381    * @param {Object=} opt_options A optional chart's configuration options.
382    * @return {!Object.<string, *>} A map of name/value pairs.
383    * @see charts.BaseChart#getOptions
384    * @private
385    */
386   function getOptions_(opt_options) {
387     opt_options = opt_options || {};
388     opt_options['stroke'] = opt_options['stroke'] || {};
389     opt_options['stroke']['color'] = opt_options['stroke']['color'] || '#ddd';
390     opt_options['stroke']['width'] = opt_options['stroke']['width'] || 1;
391     opt_options['fillcolor'] = opt_options['fillcolor'] || '#f5f5f5';
392     opt_options['rgb'] = opt_options['rgb'] || {};
393     opt_options['rgb']['red'] = 'red' in opt_options['rgb'] ?
394         opt_options['rgb']['red'] : 10;
395     opt_options['rgb']['green'] = 'green' in opt_options['rgb'] ?
396         opt_options['rgb']['green'] : 170;
397     opt_options['rgb']['blue'] = 'blue' in opt_options['rgb'] ?
398         opt_options['rgb']['blue'] : 10;
399     return self_.getOptions(opt_options);
400   }
