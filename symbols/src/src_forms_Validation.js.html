  9 /**
 10  * Simple implementation of HTML5 validation for non-supported browsers.
 11  * @constructor
 12  * @requires dom.events
 13  * @requires locale.Validation
 14  * @example
 15  * (<b>new</b> forms.Validation).init(document.forms[0]);
 16  */
 17 forms.Validation = function() {
 18 
 19   /**
 20    * @type {!Object.<string, RegExp>}
 21    * @link https://tools.ietf.org/html/rfc2822#section-3.4.1
 22    */
 23   var VALIDATORS = {
 24     'email': /^[a-z0-9]([\w\.\-\+]*[a-z0-9])*@[a-z0-9]([\w\.\-]*[a-z0-9])*$/i,
 25     'url': /^((https?|ftp):\/\/).*/
 26   };
 27 
 28   /**
 29    * Initializes HTML5 form validation feature.
 30    * @param {Node} container The HTML container which contains form elements.
 31    */
 32   this.init = function(container) {
 33     if (!isSupported_) {
 34       for (/** @type {number} */ var i = 0; i < forms.TAGS.length;) {
 35         setValidation_(dom.getElementsByTagName(container, forms.TAGS[i++]));
 36       }
 37     }
 38   };
 39 
 40   /**
 41    * @param {NodeList} elements List of HTML Form elements.
 42    * @private
 43    */
 44   function setValidation_(elements) {
 45     for (/** @type {number} */ var i = 0; i < elements.length; i++) {
 46       /** @type {Node|Element} */ var element = elements[i];
 47       initFieldValidation_(element);
 48       initFormValidation_(element.form);
 49     }
 50   }
 51 
 52   /**
 53    * @param {Node|Element} element HTML input or textarea element.
 54    * @private
 55    */
 56   function initFieldValidation_(element) {
 57     /**
 58      * A string containing the standard or custom error message that would
 59      * be displayed if the user submitted at this time.
 60      * @type {string}
 61      * @link http://msdn.microsoft.com/en-us/library/ie/hh772950
 62      */
 63     element['validationMessage'] = '';
 64 
 65     /**
 66      * Sets custom validity message.
 67      * @param {string} message String containing a custom message.
 68      * @link http://msdn.microsoft.com/en-us/library/ie/hh772949
 69      */
 70     element.setCustomValidity = function(message) {
 71       element['validationMessage'] = message;
 72     };
 73 
 74     /**
 75      * @return {boolean} Returns whether a form will validate when it is
 76      * submitted, without having to submit it.
 77      * @link http://msdn.microsoft.com/en-us/library/ie/hh772948
 78      */
 79     element.checkValidity = function() {
 80       return checkFieldValidity_(element);
 81     };
 82 
 83     if (element.hasAttribute('required')) {
 84       dom.css.addClass(element, 'required');
 85     }
 86   }
 87 
 88   /**
 89    * Initializes form validation.
 90    * @param {Node|Element} form The form element to initialize validation.
 91    * @private
 92    */
 93   function initFormValidation_(form) {
 94     if (form) {
 95       if (!form.checkValidity) {
 96         /**
 97          * @return {boolean} Returns whether a form will validate when it is
 98          * submitted, without having to submit it.
 99          * @link http://msdn.microsoft.com/en-us/library/ie/hh772948
100          */
101         form.checkValidity = function() {
102           return checkFormValidity_(form);
103         };
104       }
105 
106       if (!form.isAttached_) {
107         dom.events.addEventListener(form, dom.events.TYPE.SUBMIT, function(e) {
108           if (!form.checkValidity()) {
109             dom.events.preventDefault(e);
110             return false;
111           }
112           return true;
113         });
114 
115         form.isAttached_ = true;
116       }
117     }
118   }
119 
120   /**
121    * @param {Node|Element} element HTML input or textarea element.
122    * @return {boolean} Returns true if the form passes validation,
123    *     false otherwise.
124    * @link http://msdn.microsoft.com/en-us/library/ie/hh772948
125    * @private
126    */
127   function checkFieldValidity_(element) {
128     /** @type {string} */
129     var placeholder = element.getAttribute('placeholder') || '';
130     if (element.value == placeholder) {
131       element.value = '';
132     }
133 
134     if (element.hasAttribute('required') && !element.value) {
135       element['validationMessage'] = messages.getMessage('required');
136       element.value = placeholder;
137       return false;
138     }
139 
140     if (!checkTypeValidity_(element)) {
141       return false;
142     }
143 
144     /** @type {string} */ var pattern = element.getAttribute('pattern');
145     if (pattern && element.value) {
146       if (!(new RegExp(pattern, 'g')).test(element.value)) {
147         element['validationMessage'] = messages.getMessage('pattern');
148         return false;
149       }
150     }
151 
152     element['validationMessage'] = '';
153     if (!element.value) element.value = placeholder;
154     return true;
155   }
156 
157   /**
158    * Returns whether a form will validate when it is submitted, without having
159    * to submit it.
160    * @param {Node|Element} form The form element.
161    * @return {boolean} Returns true if the form passes validation.
162    * @link http://msdn.microsoft.com/en-us/library/ie/hh772948
163    * @private
164    */
165   function checkFormValidity_(form) {
166     /** @type {Element} */ var invalid = null;
167 
168     for (/** @type {number} */ var i = 0; i < form.elements.length; i++) {
169       /** @type {Element} */ var element = form.elements[i];
170 
171       if (element.checkValidity && !element.checkValidity()) {
172         if (!invalid)
173           invalid = element;
174       }
175 
176       if (!element.tooltip) {
177         element.tooltip = dom.createElement('SPAN');
178         dom.css.setClass(element.tooltip, 'validation-tooltip');
179         element.parentNode.insertBefore(element.tooltip, element);
180         element.parentNode.insertBefore(element, element.tooltip);
181       }
182       element.tooltip.style.display = 'none';
183     }
184 
185     if (invalid) {
186       showValidationMessage_(invalid);
187       return false;
188     }
189     return true;
190   }
191 
192   /**
193    * @param {Node|Element} element HTML input or textarea element.
194    * @return {boolean} Returns true if element value is valid, false otherwise.
195    * @private
196    */
197   function checkTypeValidity_(element) {
198     /** @type {string} */ var value = element.value;
199     /** @type {string} */ var type = element.getAttribute('type');
200     /** @type {RegExp} */ var regexp = VALIDATORS[type];
201 
202     if (value && regexp && !regexp.test(value)) {
203       element['validationMessage'] = messages.getMessage(type);
204       return false;
205     }
206     return true;
207   }
208 
209   /**
210    * @param {Node|Element} element HTML element.
211    * @private
212    */
213   function showValidationMessage_(element) {
214     if (!showValidationMessage_.isAttached_) {
215       dom.events.addEventListener(
216           dom.document, dom.events.TYPE.MOUSEDOWN, function() {
217             if (invalidElement_ && invalidElement_.tooltip) {
218               invalidElement_.tooltip.style.display = 'none';
219             }
220           });
221 
222       showValidationMessage_.isAttached_ = true;
223     }
224     element.tooltip.innerHTML = element['validationMessage'];
225     element.tooltip.style.display = 'block';
226     element.focus();
227     invalidElement_ = element;
228   }
229 
230   /**
231    * @type {boolean}
232    * @private
233    */
234   var isSupported_ = 'required' in dom.createElement('INPUT');
235 
236   /**
237    * @type {Node|Element}
238    * @private
239    */
240   var invalidElement_ = null;
241 
242   /**
243    * @type {!locale.Validation}
244    * @private
245    */
246   var messages = new locale.Validation;
247 };
