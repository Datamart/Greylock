 10 /**
 11  * Creates and return a URL object composed from the given parameters.
 12  * @param {string} url Is a String representing an absolute or relative URL.
 13  * @constructor
 14  */
 15 net.URL = window['URL'] || function(url) {
 16   /** @type {!RegExp} */
 17   var pattern = /^(([^:\/?#]+):)?(\/\/([^\/?#]*))?([^?#]*)(\?([^#]*))?(#(.*))?/;
 18   /** @type {Array} */
 19   var result = (url || '').match(pattern) || [];
 20 
 21   this['protocol'] = result[2] ? result[2] + ':' : '';
 22   this['host'] = result[4] || '';
 23   this['hostname'] = this['host'].split(':')[0];
 24   this['port'] = +(this['host'].split(':')[1]) || '';
 25   this['pathname'] = result[5] || '';
 26   this['search'] = result[7] ? ('?' + result[7]) : '';
 27   this['hash'] = result[9] ? ('#' + result[9]) : '';
 28 
 29   this['toString'] = function() {
 30     return this['protocol'] + '//' + this['host'] +
 31            this['pathname'] + this['search'] + this['hash'];
 32   };
 33   return this;
 34 };
