 10 /**
 11  * Constructor of PropertyReader.
 12  * @constructor
 13  */
 14 util.PropertyReader = function() {
 15 
 16   /**
 17    * Gets attribute' values of "tagName" founded in "context".
 18    * @param {Document|Node} context The DOM context.
 19    * @param {string} tagName The element tag name.
 20    * @param {string} attribute The name of attribute.
 21    * @param {string} value The name of value attribute.
 22    */
 23   this.read = function(context, tagName, attribute, value) {
 24     if (context) {
 25       /** @type {NodeList} */
 26       var nodes = dom.getElementsByTagName(context, tagName);
 27       for (/** @type {number} */ var i = 0; i < nodes.length;) {
 28         /** @type {!Node} */ var node = nodes[i++];
 29         /** @type {string} */ var name = node.getAttribute(attribute);
 30         if (name) {
 31           /** @type {string} */ var result = node.getAttribute(value);
 32           cache_[name] = result;
 33           values_.push(result);
 34           names_.push(name);
 35         }
 36       }
 37     }
 38   };
 39 
 40   /**
 41    * TODO: rename to 'getValues'
 42    * @return {!Array.<string>} Returns object.
 43    */
 44   this.getListOfValues = function() {
 45     return values_;
 46   };
 47 
 48   /**
 49    * @return {!Array.<string>} Returns object.
 50    */
 51   this.getNames = function() {
 52     return names_;
 53   };
 54 
 55   /**
 56    * @return {Object.<string, string>} Returns object.
 57    */
 58   this.getData = function() {
 59     return cache_;
 60   };
 61 
 62   /**
 63    * @param {string} key The property key.
 64    * @return {?string} Returns property value.
 65    */
 66   this.getValue = function(key) {
 67     return cache_[key];
 68   };
 69 
 70   /**
 71    * @dict
 72    * @private
 73    */
 74   var cache_ = {};
 75 
 76   /**
 77    * @type {!Array.<string>}
 78    * @private
 79    */
 80   var values_ = [];
 81 
 82   /**
 83    * @type {!Array.<string>}
 84    * @private
 85    */
 86   var names_ = [];
 87 };
