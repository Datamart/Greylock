  9 /**
 10  * Miscellaneous Cookies utility methods.
 11  * @type {!Object.<string, Function>}
 12  */
 13 dom.Cookies = {
 14   /**
 15    * Sets a cookie.
 16    * @param {string} key The name of the cookie.
 17    * @param {string} value The value of the cookie.
 18    * @param {number} expires The days after the cookie expires.
 19    * @param {string=} opt_domain Optional, domain that the cookie is
 20    *     available to.
 21    */
 22   set: function(key, value, expires, opt_domain) {
 23     dom.document.cookie = escape(key) + '=' + escape(value || '') +
 24         '; expires=' + (new Date(+new Date + (expires * 864e5)).toGMTString()) +
 25         '; path=/; domain=' + (opt_domain || dom.document.domain);
 26   },
 27 
 28   /**
 29    * Gets the value for the first cookie with the given name.
 30    * @param {string} key The name of the cookie to get.
 31    * @param {string=} opt_default The optional default value.
 32    * @return {string} The value of the cookie. If no cookie is set this
 33    *     returns opt_default or undefined if opt_default is not provided.
 34    */
 35   get: function(key, opt_default)  {
 36     return unescape(
 37         (dom.document.cookie.match(key + '=([^;].+?)(;|$)') || [])[1] ||
 38         opt_default || '');
 39   },
 40 
 41   /**
 42    * Removes and expires a cookie.
 43    * @param {string} key The cookie name.
 44    * @return {boolean} Whether the cookie existed before it was removed.
 45    */
 46   remove: function(key) {
 47     /** @type {string} */ var value = dom.Cookies.get(key);
 48     dom.Cookies.set(key, '', 0);
 49     return !!value;
 50   },
 51 
 52   /**
 53    * Removes and expires all cookie.
 54    */
 55   clear: function() {
 56     /** @type {!Array.<string>} */ var keys = dom.Cookies.keys();
 57     while (keys.length) {
 58       dom.Cookies.remove(keys.pop());
 59     }
 60   },
 61 
 62   /**
 63    * Gets list of stored keys.
 64    * @return {!Array.<string>} Returns list of stored keys.
 65    */
 66   keys: function() {
 67     /** @type {!Array.<string>} */ var keys = [];
 68     /** @type {!RegExp} */ var re = /;\s*/;
 69     /** @type {!Array.<string>} */ var parts = dom.document.cookie.split(re);
 70     while (parts.length) {
 71       keys.push(parts.pop().split('=')[0]);
 72     }
 73     return keys;
 74   }
 75 };
