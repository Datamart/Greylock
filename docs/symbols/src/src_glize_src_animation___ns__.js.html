 16 /**
 17  * Runs animations.
 18  * @param {!Node} element The element to animate.
 19  * @param {!Object.<string, number>} options Animation options.
 20  * @static
 21  * @see charts.BarChart
 22  * @see charts.ColumnChart
 23  * @example
 24  * animation.animate(element, {'width': 500});
 25  * animation.animate(element, {'width': 0});
 26  */
 27 animation.animate = function(element, options) {
 28   // Testing.
 29   // if (element.tagName == 'DIV') return animation.run_(element, options);
 30 
 31   if ('transition' in element.style) {
 32     /** @type {!RegExp} */ var isDigit = /^\d+([\.]{1,}\d+)?$/;
 33     for (/** @type {string} */ var key in options) {
 34       // 99ms fix Firefox delay issue.
 35       /** @type {number} */ var delay = window['XULElement'] ? 99 : 0;
 36       (function(prop) {
 37         /** @type {number} */ var timer = setTimeout(function() {
 38           element.style.transition = 'all ' +
 39               Math.random().toFixed(2) + 's ease-in-out';
 40           if ('transform' == prop) {
 41             element.style.transform = options[prop];
 42             element.style.WebkitTransform = options[prop];
 43             element.style.MozTransform = options[prop];
 44             element.style.OTransform = options[prop];
 45           } else {
 46             element.style[prop] = options[prop] +
 47                 (isDigit.test(options[prop]) ? 'px' : '');
 48           }
 49           clearTimeout(timer);
 50         }, delay);
 51       })(key);
 52     }
 53   } else {
 54     animation.run_(element, options);
 55   }
 56 };

 59 /**
 60  * Runs animation by property.
 61  * @param {Node} element Animated element.
 62  * @param {Object.<string, number>} options Animation options.
 63  * @private
 64  */
 65 animation.run_ = function(element, options) {
 66   /**
 67    * @param {function(number)} frame The frame function.
 68    * @param {function(number): number=} opt_delta The optional delta function.
 69    * @param {number=} opt_duration The optional duration.
 70    * @param {number=} opt_delay The optional delay.
 71    */
 72   function animate(frame, opt_delta, opt_duration, opt_delay) {
 73     opt_duration = (opt_duration || 300) * Math.random() + 100;
 74     opt_delay = opt_delay || 10;
 75     opt_delta = opt_delta || function(progress) {
 76       //return Math.pow(progress, 2);
 77       return progress;
 78     };
 79 
 80     /** @type {!Date} */ var start = new Date;
 81     /** @type {number} */ var interval = setInterval(function() {
 82       /** @type {number} */ var progress = (new Date - start) / opt_duration;
 83       if (progress > 1) progress = 1;
 84       frame(opt_delta(progress));
 85       if (progress == 1) clearInterval(interval);
 86     }, opt_delay);
 87   }
 88 
 89   animate(function(delta) {
 90     for (/** @type {string} */ var prop in options) {
 91       element.style[prop] = options[prop] * delta + 'px';
 92     }
 93   });
 94 };
