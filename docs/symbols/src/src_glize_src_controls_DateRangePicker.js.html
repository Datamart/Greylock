 10 /**
 11  * Constructor of DateRangePicker.
 12  * @param {Object=} opt_options Optional options.
 13  * @extends {controls.DatePicker} controls.DatePicker
 14  * @constructor
 15  */
 16 controls.DateRangePicker = function(opt_options) {
 17   opt_options = opt_options || {};
 18   opt_options['multiple'] = true;
 19   opt_options['separator'] = opt_options['separator'] || ' - ';
 20   opt_options['timeout'] = opt_options['timeout'] || 300;
 21 
 22   controls.DatePicker.apply(this, [opt_options]);
 23 
 24   /** @inheritDoc */
 25   this.clickHandler = function() {
 26     /** @type {controls.Calendar} */ var calendar = self_.getCalendar();
 27     /** @type {Array.<Date>} */ var dates = calendar.getDates();
 28     /** @type {number} */ var length = dates.length;
 29     if (length > 1) {
 30       /** @type {string} */ var format = self_.getFormat();
 31       /** @type {Date} */ var start = dates[0];
 32       /** @type {Date} */ var end = dates[length - 1];
 33       self_.setValue(
 34           formatters.DateFormatter.format(start, format) +
 35           opt_options['separator'] +
 36           formatters.DateFormatter.format(end, format));
 37 
 38       setTimeout(self_.hide, opt_options['timeout']);
 39     }
 40   };
 41 
 42   /** @inheritDoc */
 43   this.getDates = function() {
 44     /** @type {!Array.<Date>} */ var range = [];
 45     /** @type {Date} */ var start = new Date;
 46     /** @type {Date} */ var end = new Date;
 47 
 48     /** @type {Array.<string>} */
 49     var value = self_.getValue().split(opt_options['separator']);
 50     /** @type {number} */ var length = value.length;
 51     if (length == 2) {
 52       /** @type {string} */ var format = self_.getFormat();
 53       start = formatters.DateFormatter.parse(value[0], format);
 54       end = formatters.DateFormatter.parse(value[length - 1], format);
 55     }
 56 
 57     while (start <= end) {
 58       range.push(new Date(start));
 59       start.setDate(start.getDate() + 1);
 60       //start = new Date(+start + 864e5);
 61     }
 62     return range;
 63   };
 64 
 65   /**
 66    * The reference to current class instance. Used in private methods.
 67    * @type {!controls.DateRangePicker}
 68    * @private
 69    */
 70   var self_ = this;
 71 
 72   /**
 73    * Indicates first click, selecting start date.
 74    * @type {boolean}
 75    * @private
 76    */
 77   var first_ = true;
 78 };

 81 /**
 82  * @param {Node|Element} element The related element.
 83  * @static
 84  */
 85 controls.DateRangePicker.show = function(element) {
 86   if (!element.picker_) {
 87     element.picker_ = new controls.DateRangePicker;
 88   }
 89   element.picker_.show(element);
 90 };

 92 // Export for closure compiler.
 93 controls.DateRangePicker['show'] = controls.DateRangePicker.show;
