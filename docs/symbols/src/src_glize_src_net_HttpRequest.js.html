 10 /**
 11  * The constructor initiates a XMLHttpRequest.
 12  * @constructor
 13  */
 14 net.HttpRequest = function() {
 15   /**
 16    * Performs DELETE request.
 17    * @param {string} url The URL to which to send the request.
 18    * @param {function(XMLHttpRequest)} callback A JavaScript function object
 19    *     that is called whenever the <code>readyState</code> attribute equals
 20    *     to 4 (DONE).
 21    * @return {!XMLHttpRequest} Returns instance of XMLHttpRequest.
 22    */
 23   this.doDelete = function(url, callback) {
 24     return doRequest_('DELETE', url, callback);
 25   };
 26 
 27   /**
 28    * Performs GET request.
 29    * @param {string} url The URL to which to send the request.
 30    * @param {function(XMLHttpRequest)} callback A JavaScript function object
 31    *     that is called whenever the <code>readyState</code> attribute equals
 32    *     to 4 (DONE).
 33    * @return {!XMLHttpRequest} Returns instance of XMLHttpRequest.
 34    */
 35   this.doGet = function(url, callback) {
 36     // @link http://tools.ietf.org/html/rfc2616#section-3.2.1
 37     if (url.length > 255) {
 38       /** @type {!Array} */ var parts = url.split('?');
 39       return self_.doPost(parts[0], callback, parts[1]);
 40     }
 41     return doRequest_('GET', url, callback);
 42   };
 43 
 44   /**
 45    * Performs HEAD request.
 46    * @param {string} url The URL to which to send the request.
 47    * @param {function(XMLHttpRequest)} callback A JavaScript function object
 48    *     that is called whenever the <code>readyState</code> attribute equals
 49    *     to 4 (DONE).
 50    * @return {!XMLHttpRequest} Returns instance of XMLHttpRequest.
 51    */
 52   this.doHead = function(url, callback) {
 53     return doRequest_('HEAD', url, callback);
 54   };
 55 
 56   /**
 57    * Performs POST request.
 58    * @param {string} url The URL to which to send the request.
 59    * @param {function(XMLHttpRequest)} callback A JavaScript function object
 60    *     that is called whenever the <code>readyState</code> attribute equals
 61    *     to 4 (DONE).
 62    * @param {Object|string} data Provides the request entity body.
 63    * @return {!XMLHttpRequest} Returns instance of XMLHttpRequest.
 64    */
 65   this.doPost = function(url, callback, data) {
 66     return doRequest_('POST', url, callback, data);
 67   };
 68 
 69   /**
 70    * Performs PUT request.
 71    * @param {string} url The URL to which to send the request.
 72    * @param {function(XMLHttpRequest)} callback A JavaScript function object
 73    *     that is called whenever the <code>readyState</code> attribute equals
 74    *     to 4 (DONE).
 75    * @param {Object|string} data Provides the request entity body.
 76    * @return {!XMLHttpRequest} Returns instance of XMLHttpRequest.
 77    */
 78   this.doPut = function(url, callback, data) {
 79     return doRequest_('PUT', url, callback, data);
 80   };
 81 
 82   /**
 83    * Gets count of active requests.
 84    * @return {number} Returns count of active requests.
 85    */
 86   this.getCount = function() {
 87     return net.HttpRequest.count_ || 0;
 88   };
 89 
 90   /**
 91    * Performs HTTP request.
 92    * @param {string} method The HTTP method to use, such as "GET", "POST",
 93    *     "PUT", "DELETE", etc. Ignored for non-HTTP(S) URLs.
 94    * @param {string} url The URL to which to send the request.
 95    * @param {function(XMLHttpRequest)} callback A JavaScript function object
 96    *     that is called whenever the <code>readyState</code> attribute equals
 97    *     to 4 (DONE).
 98    * @param {string|Object=} opt_data Provides the request entity body.
 99    *     If opt_data is Object, request will be sent as application/json.
100    * @return {!XMLHttpRequest} Returns instance of XMLHttpRequest.
101    * @private
102    */
103   function doRequest_(method, url, callback, opt_data) {
104     url = url.replace(/^\/+/, '/');
105     url += (~url.indexOf('?') ? '&' : '?') + 'nocache=' + (+new Date);
106     net.HttpRequest.count_++;
107 
108     /** @type {!XMLHttpRequest} */
109     var req = window.XMLHttpRequest ? new XMLHttpRequest :
110                                       new ActiveXObject('Microsoft.XMLHTTP');
111     req.onreadystatechange = function() {
112       if (req.readyState == 4) {
113         net.HttpRequest.count_--;
114         callback(req);
115       }
116     };
117     req.open(method, url, true);
118     req.send(prepare_(req, method, opt_data));
119     return req;
120   }
121 
122   /**
123    * Sets request <code>request</code> headers depending to request
124    * <code>method</code> and <code>opt_data</code>.
125    * @param {!XMLHttpRequest} request The instance of XMLHttpRequest.
126    * @param {string} method The HTTP method to use, such as "GET", "POST",
127    *     "PUT", "DELETE", etc. Ignored for non-HTTP(S) URLs.
128    * @param {string|Object=} opt_data Provides the request entity body.
129    *     If opt_data is Object, request will be sent as application/json.
130    * @return {string} Returns modified <code>opt_data</code>.
131    * @private
132    */
133   function prepare_(request, method, opt_data) {
134     for (/** @type {string} */ var name in net.HttpRequest.headers_) {
135       request.setRequestHeader(name, net.HttpRequest.headers_[name]);
136     }
137 
138     request.setRequestHeader('X-Requested-With', 'XMLHttpRequest');
139 
140     if (opt_data && typeof opt_data == 'object') {
141       request.setRequestHeader('Content-Type',
142           'application/json; charset=' + dom.CHARSET);
143       opt_data = util.StringUtils.JSON.stringify(
144           /** @type {!Object} */ (opt_data));
145     } else if (method == 'POST') {
146       request.setRequestHeader('Content-Type',
147           'application/x-www-form-urlencoded; charset=' + dom.CHARSET);
148     }
149 
150     return /** @type {string} */ (opt_data);
151   }
152 
153   /**
154    * The reference to current class instance.
155    * Used in private methods and for preventing jslint errors.
156    * @type {!net.HttpRequest}
157    * @private
158    */
159   var self_ = this;
160 
161   /**
162    * The count of active requests.
163    * @static
164    * @type {number}
165    * @see net.HttpRequest#getCount
166    * @private
167    */
168   net.HttpRequest.count_ = net.HttpRequest.count_ || 0;
169 };
