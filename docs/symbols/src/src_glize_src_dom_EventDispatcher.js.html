 11 /**
 12  * The EventDispatcher class implements W3C EventTarget and EventListener
 13  * interfaces.
 14  * @constructor
 15  * @example
 16  * <b>function</b> UserMenu() {
 17  *    dom.EventDispatcher.call(<b>this</b>);
 18  *
 19  *   <b>this</b>.onClick = <b>function</b>() {
 20  *     <b>this</b>.dispatchEvent('menu.clicked');
 21  *   };
 22  * }
 23  *
 24  * <b>function</b> ContentArea() {
 25  *    dom.EventDispatcher.call(<b>this</b>);
 26  *
 27  *    <b>function</b> update_(menu) {
 28  *      console.log(menu);
 29  *    }
 30  *
 31  *    <b>this</b>.addEventListener('menu.clicked', update_);
 32  * }
 33  */
 34 dom.EventDispatcher = function() {
 35 
 36   /**
 37    * Registrates event listener on the event target.
 38    * @param {string} type The event type for which the user is registering.
 39    * @param {function(dom.EventDispatcher)} listener The listener parameter
 40    *     takes an interface implemented by the user which contains the
 41    *     methods to be called when the event occurs.
 42    */
 43   this.addEventListener = function(type, listener) {
 44     events_[type] = (events_[type] || []).concat([listener]);
 45   };
 46 
 47   /**
 48    * Removes event listeners from the event target.
 49    * @param {string} type The event type of the listener being removed.
 50    * @param {function(dom.EventDispatcher)} listener Reference to the event
 51    *     listener to be removed.
 52    * @return {boolean} Returns true if listener was removed.
 53    */
 54   this.removeEventListener = function(type, listener) {
 55     /** @type {Array} */ var listeners = events_[type];
 56     if (listeners) {
 57       /** @type {number} */ var index = listeners.length >>> 0;
 58       while (index--) {
 59         if (listeners[index] === listener) {
 60           listeners.splice(index, 1);
 61           return true;
 62         }
 63       }
 64     }
 65     return false;
 66   };
 67 
 68   /**
 69    * Dispatches events into the implementations event model.
 70    * @param {string} type The event type to be used in processing the event.
 71    */
 72   this.dispatchEvent = function(type) {
 73     /** @type {Array} */ var listeners = events_[type];
 74     if (listeners) {
 75       /** @type {number} */ var index = 0;
 76       while (index < listeners.length) {
 77         listeners[index++](self_);
 78       }
 79     }
 80   };
 81 
 82   /**
 83    * Event storage.
 84    * @type {!Object.<string, Array>}
 85    * @private
 86    */
 87   var events_ = {};
 88 
 89   /**
 90    * The reference to current class instance.
 91    * Used in private methods and for preventing jslint errors.
 92    * @type {!dom.EventDispatcher}
 93    * @private
 94    */
 95   var self_ = this;
 96 };
