 10 /**
 11  * Constructor of DatePicker.
 12  * @param {Object=} opt_options Optional options.
 13  * @constructor
 14  * @requires formatters.DateFormatter
 15  * @requires controls.Calendar
 16  * @example
 17  * <input onclick="controls.DatePicker.show(this)"
 18  *        data-format="YYYY/MM/dd"
 19  *        value="2013/06/20"
 20  *        readonly>
 21  * <input onclick="controls.DatePicker.show(this)"
 22  *        data-format="MMM dd YYYY"
 23  *        value="May 20 2013"
 24  *        readonly>
 25  * <style>
 26  * table.calendar {border: solid 1px gray; border-collapse: collapse;
 27  *                 background: white; font-family: Arial; font-size: 13px;}
 28  * table.calendar thead {background: silver;}
 29  * table.calendar td,
 30  * table.calendar th {border: solid 1px gray;}
 31  * table.calendar td.today {font-weight: bold; background: silver;}
 32  * table.calendar td.active {font-weight: bold; background: blue; color: white;}
 33  * table.calendar th.prev,
 34  * table.calendar th.next {cursor: pointer;}
 35  * </style>
 36  * <script src="../../bin/jscb.js"></script>
 37  * <input onclick="controls.DatePicker.show(this)"
 38  *        data-format="YYYY/MM/dd" value="2013/06/20" readonly>
 39  */
 40 controls.DatePicker = function(opt_options) {
 41 
 42   opt_options = opt_options || {};
 43   opt_options['selectable'] = true;
 44   opt_options['format'] = opt_options['format'] || 'YYYY-MM-dd';
 45 
 46   /**
 47    * Gets date format.
 48    * <br>Gets date format from element <code>data-format</code> attribute,
 49    * otherwise returns <code>YYYY-MM-dd</code>.
 50    * @return {string} Returns date format.
 51    */
 52   this.getFormat = function() {
 53     return element_.getAttribute('data-format') || opt_options['format'];
 54   };
 55 
 56   /**
 57    * <code>protected</code> Gets related HTML element.
 58    * @return {Node} Returns related HTML element.
 59    * @protected
 60    */
 61   this.getElement = function() {
 62     return element_;
 63   };
 64 
 65   /**
 66    * <code>protected</code> Gets reference to
 67    * <code>{@link controls.Calendar}</code>.
 68    * @return {controls.Calendar} Returns reference to
 69    * <code>{@link controls.Calendar}</code>.
 70    * @protected
 71    */
 72   this.getCalendar = function() {
 73     return calendar_;
 74   };
 75 
 76   /**
 77    * Shows date picker control.
 78    * @param {Node|Element} element The related element.
 79    * @example
 80    * <input onclick="controls.DatePicker.show(this)"
 81    *        data-format="YYYY/MM/dd"
 82    *        value="2013/06/20"
 83    *        readonly>
 84    */
 85   this.show = function(element) {
 86     /** @type {Node} */ var picker = controls.DatePicker.control_;
 87     if (picker.style.display == 'block') {
 88       self_.hide();
 89     } else {
 90       element_ = element;
 91       draw_();
 92       /** @type {number} */ var y = dom.document.documentElement.scrollTop ||
 93                                     window.pageYOffset || 0;
 94       /** @type {number} */ var x = dom.document.documentElement.scrollLeft ||
 95                                     window.pageXOffset || 0;
 96       /** @type {Object} */ var rect = dom.getBoundingClientRect(element);
 97       // TODO: implement possibility to show picker above element.
 98       picker.style.top = (rect['bottom'] + y) + 'px';
 99       picker.style.left = (rect['left'] + x) + 'px';
100       picker.style.display = 'block';
101       dom.events.addEventListener(
102           dom.document, dom.events.TYPE.KEYDOWN, keydown_);
103       dom.events.addEventListener(
104           dom.document, dom.events.TYPE.MOUSEDOWN, mousedown_);
105     }
106   };
107 
108   // Export for closure compiler.
109   this['show'] = this.show;
110 
111   /**
112    * Hides date picker control.
113    */
114   this.hide = function() {
115     if (controls.DatePicker.control_)
116       controls.DatePicker.control_.style.display = 'none';
117     dom.events.removeEventListener(
118         dom.document, dom.events.TYPE.KEYDOWN, keydown_);
119     dom.events.removeEventListener(dom.document,
120         dom.events.TYPE.MOUSEDOWN, mousedown_);
121   };
122 
123   // Export for closure compiler.
124   this['hide'] = this.hide;
125 
126   /**
127    * Sets formatted date as element value.
128    * <br><i>Note:</i> For HTML form elements sets <code>value</code> attribute,
129    * for other HTML elements sets <code>innerHTML</code>.
130    * Dispatches {@link dom.events.TYPE.CHANGE} event.
131    * @param {string} value The formatted date value.
132    */
133   this.setValue = function(value) {
134     if ('value' in element_) element_.value = value;
135     else element_.innerHTML = value;
136     dom.events.dispatchEvent(element_, dom.events.TYPE.CHANGE);
137   };
138 
139   /**
140    * Gets element value.
141    * <br><i>Note:</i> From HTML form elements gets <code>value</code> attribute,
142    * from other HTML elements gets <code>textContent || innerText</code>.
143    * @return {string} Returns element value.
144    */
145   this.getValue = function() {
146     return element_.getAttribute('data-value') || element_.value ||
147            element_.textContent || element_.innerText;
148   };
149 
150   /**
151    * <code>protected</code> Gets list of date objects.
152    * @return {!Array.<Date>} Returns list of date objects.
153    * @protected
154    */
155   this.getDates = function() {
156     /** @type {string} */ var value = self_.getValue();
157     return [value ? formatter_.parse(value, self_.getFormat()) : new Date];
158   };
159 
160   /**
161    * Handles calendar click event.
162    * @protected
163    */
164   this.clickHandler = function() {
165     self_.setValue(formatter_.format(calendar_.getDate(), self_.getFormat()));
166     self_.hide();
167   };
168 
169   /**
170    * @private
171    */
172   function draw_() {
173     calendar_ ? calendar_.clear() : init_();
174     calendar_.draw(self_.getDates());
175     calendar_.addEventListener(dom.events.TYPE.CLICK, self_.clickHandler);
176   }
177 
178   /**
179    * @private
180    */
181   function init_() {
182     if (!controls.DatePicker.control_) {
183       controls.DatePicker.control_ = dom.document.body.appendChild(
184           dom.createElement('DIV'));
185       controls.DatePicker.control_.style.position = 'absolute';
186       dom.css.setClass(controls.DatePicker.control_, 'date-picker');
187     }
188     calendar_ = new controls.Calendar(
189         controls.DatePicker.control_, opt_options);
190   }
191 
192   /**
193    * @param {Event} e The keydown event.
194    * @private
195    */
196   function keydown_(e) {
197     e = e || window.event;
198     /** @type {number} */ var code = e.keyCode || e.which;
199     if (27 == code) {
200       self_.hide();
201     }
202   }
203 
204   /**
205    * @param {Event} e The mousedown event.
206    * @private
207    */
208   function mousedown_(e) {
209     e = e || window.event;
210     /** @type {boolean} */ var hide = true;
211     /** @type {Node|Element} */ var target = e.target || e.srcElement;
212     if (target) {
213       while (target && target.tagName != 'BODY') {
214         if (target == controls.DatePicker.control_) {
215           hide = false;
216           break;
217         }
218         target = target.parentNode;
219       }
220       if (hide) self_.hide();
221     }
222   }
223 
224   /**
225    * The reference to current class instance. Used in private methods.
226    * @type {!controls.DatePicker}
227    * @private
228    */
229   var self_ = this;
230 
231   /**
232    * @type {Node}
233    * @private
234    */
235   var element_ = null;
236 
237   /**
238    * @type {controls.Calendar}
239    * @private
240    */
241   var calendar_ = null;
242 
243   /**
244    * @type {!formatters.DateFormatter}
245    * @private
246    */
247   var formatter_ = new formatters.DateFormatter;
248 
249   init_();
250 };

253 /**
254  * Shows date picker control.
255  * @param {Node} element The related element.
256  * @static
257  */
258 controls.DatePicker.show = function(element) {
259   if (!element.picker_) {
260     element.picker_ = new controls.DatePicker;
261   }
262   element.picker_.show(element);
263 };
