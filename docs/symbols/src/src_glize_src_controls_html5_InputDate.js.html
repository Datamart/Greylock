 10 /**
 11  * Constructor of InputDate.<br>
 12  * The following 8 pseudo-elements are made available by WebKit for customizing
 13  * a date inputâ€™s textbox:<code><pre>
 14  *   ::-webkit-datetime-edit { padding: 1em; }
 15  *   ::-webkit-datetime-edit-fields-wrapper { background: silver; }
 16  *   ::-webkit-datetime-edit-text { color: red; padding: 0 0.3em; }
 17  *   ::-webkit-datetime-edit-month-field { color: blue; }
 18  *   ::-webkit-datetime-edit-day-field { color: green; }
 19  *   ::-webkit-datetime-edit-year-field { color: purple; }
 20  *   ::-webkit-inner-spin-button { display: none; }
 21  *   ::-webkit-calendar-picker-indicator { background: orange; }
 22  * </pre></code>
 23  * @param {string|Element} input The input element or its ID attribute.
 24  * @constructor
 25  * @requires controls.DatePicker
 26  * @requires formatters.DateFormatter
 27  * @link http://www.w3.org/TR/html-markup/input.date.html
 28  */
 29 controls.html5.InputDate = function(input) {
 30   // http://trac.webkit.org/wiki/Styling%20Form%20Controls
 31   // There are no ways to specify the date format in the text box. It always
 32   // reflects OS setting. Also, there are no ways to styling the native
 33   // calendar picker.
 34 
 35   /**
 36    * @private
 37    */
 38   function init_() {
 39     if ('date' != input_.type) {
 40       picker_ = new controls.DatePicker;
 41       dom.events.addEventListener(input_, dom.events.TYPE.CLICK, mousedown_);
 42       dom.events.addEventListener(input_, dom.events.TYPE.CHANGE, change_);
 43       input_.readOnly = true;
 44 
 45       // -moz-appearance: menulist;
 46       // -webkit-appearance: menulist;
 47 
 48       //input_.style.backgroundImage = 'url(data:image/png;base64,' +  + ')';
 49       //input_.style.backgroundPosition = 'right center';
 50       //input_.style.backgroundRepeat = 'no-repeat';
 51       //input_.style.paddingRight = PADDING + 'px';
 52     }
 53   }
 54 
 55   /**
 56    * @param {Event} e The mousedown event.
 57    * @private
 58    */
 59   function mousedown_(e) {
 60     picker_.show(input_);
 61   }
 62 
 63   /**
 64    * @param {Event} e The onchange event.
 65    * @private
 66    */
 67   function change_(e) {
 68     /** @type {string} */ var format = picker_.getFormat();
 69     /** @type {Date} */
 70     var value = formatters.DateFormatter.parse(input_.value, format);
 71     if (value) {
 72       if (max_) {
 73         /** @type {Date} */
 74         var max = formatters.DateFormatter.parse(max_, format);
 75         if (value > max) {
 76           value = max;
 77         }
 78       }
 79 
 80       if (min_) {
 81         /** @type {Date} */
 82         var min = formatters.DateFormatter.parse(min_, format);
 83         if (value < min) {
 84           value = min;
 85         }
 86       }
 87 
 88       input_.value = formatters.DateFormatter.format(value, format);
 89     }
 90   }
 91 
 92   /**
 93    * @type {controls.DatePicker}
 94    * @private
 95    */
 96   var picker_ = null;
 97 
 98   /**
 99    * The reference to input element.
100    * @type {Element}
101    * @private
102    */
103   var input_ = typeof input == 'string' ? dom.getElementById(input) : input;
104 
105   /**
106    * The expected lower bound for the input value.
107    * A valid full-date as defined in [RFC 3339].
108    * @link http://tools.ietf.org/html/rfc3339
109    * @type {string}
110    * @private
111    */
112   var min_ = input_.getAttribute('min') || '';
113 
114   /**
115    * The expected upper bound for the input value.
116    * A valid full-date as defined in [RFC 3339].
117    * @link http://tools.ietf.org/html/rfc3339
118    * @type {string}
119    * @private
120    */
121   var max_ = input_.getAttribute('max') || '';
122 
123   /**
124    * Specifies the value granularity of the input value.
125    * Currently not used.
126    * @type {number}
127    * @private
128    */
129   var step_ = +(input_.getAttribute('step') || 1);
130 
131   init_();
132 };
