  9 /**
 10  * Simple implementation of HTML5 validation for non-supported browsers.
 11  * @constructor
 12  * @requires dom.events
 13  * @requires locale.Validation
 14  * @example
 15  * (<b>new</b> forms.Validation).init(document.forms[0]);
 16  */
 17 forms.Validation = function() {
 18 
 19   /**
 20    * @type {!Object.<string, RegExp>}
 21    * @link https://tools.ietf.org/html/rfc2822#section-3.4.1
 22    */
 23   var VALIDATORS = {
 24     'email': /^[a-z0-9]([\w\.\-\+]*[a-z0-9])*@[a-z0-9]([\w\.\-]*[a-z0-9])*$/i,
 25     'url': /^((https?|ftp):\/\/).+/,
 26     'number': /^-?\d+(.\d+)?(e-?\d+)?$/i
 27   };
 28 
 29   /**
 30    * Initializes HTML5 form validation feature.
 31    * @param {Node} container The HTML container which contains form elements.
 32    */
 33   this.init = function(container) {
 34     if (!isSupported_()) {
 35       for (/** @type {number} */ var i = 0; i < forms.TAGS.length;) {
 36         setValidation_(dom.getElementsByTagName(container, forms.TAGS[i++]));
 37       }
 38     }
 39   };
 40 
 41   /**
 42    * @param {NodeList} elements List of HTML Form elements.
 43    * @private
 44    */
 45   function setValidation_(elements) {
 46     for (/** @type {number} */ var i = 0; i < elements.length; i++) {
 47       /** @type {Node|Element} */ var element = elements[i];
 48       initFieldValidation_(element);
 49       initFormValidation_(element.form);
 50     }
 51   }
 52 
 53   /**
 54    * @param {Node|Element} element HTML input or textarea element.
 55    * @private
 56    */
 57   function initFieldValidation_(element) {
 58     if (!element.checkValidity) {
 59       /**
 60        * @return {boolean} Returns whether a form will validate when it is
 61        * submitted, without having to submit it.
 62        * @link http://msdn.microsoft.com/en-us/library/ie/hh772948
 63        */
 64       element.checkValidity = function() {
 65         return checkFieldValidity_(element);
 66       };
 67     }
 68 
 69     if (!element.setCustomValidity) {
 70       /**
 71        * Sets custom validity message.
 72        * @param {string} message String containing a custom message.
 73        * @link http://msdn.microsoft.com/en-us/library/ie/hh772949
 74        * @link http://msdn.microsoft.com/en-us/library/ie/hh772950
 75        */
 76       element.setCustomValidity = function(message) {
 77         element['validationMessage'] = message;
 78       };
 79     }
 80 
 81     element.setCustomValidity('');
 82     if (element.hasAttribute('required')) {
 83       dom.css.addClass(element, 'required');
 84     }
 85   }
 86 
 87   /**
 88    * Initializes form validation.
 89    * @param {Node|Element} form The form element to initialize validation.
 90    * @private
 91    */
 92   function initFormValidation_(form) {
 93     if (form) {
 94       if (!form.checkValidity || isSafari_()) {
 95         /**
 96          * @return {boolean} Returns whether a form will validate when it is
 97          * submitted, without having to submit it.
 98          * @link http://msdn.microsoft.com/en-us/library/ie/hh772948
 99          */
100         form.checkValidity = function() {
101           return checkFormValidity_(form);
102         };
103       }
104 
105       if (!form.isAttached_) {
106         dom.events.addEventListener(form, dom.events.TYPE.SUBMIT, function(e) {
107           if (!form.checkValidity()) {
108             dom.events.preventDefault(e);
109             return false;
110           }
111           return true;
112         });
113 
114         form.isAttached_ = true;
115       }
116     }
117   }
118 
119   /**
120    * @param {Node|Element} element HTML input or textarea element.
121    * @return {boolean} Returns true if the form passes validation,
122    *     false otherwise.
123    * TODO: Add attribute validation (min, max, step, etc.)
124    * @link http://msdn.microsoft.com/en-us/library/ie/hh772948
125    * @private
126    */
127   function checkFieldValidity_(element) {
128     /** @type {string} */
129     var placeholder = element.getAttribute('placeholder') || '';
130     if (element.value == placeholder) {
131       element.value = '';
132     }
133 
134     if (element.hasAttribute('required') && !element.value) {
135       element.setCustomValidity(messages.getMessage('required'));
136       if (!forms.isSupported(forms.FEATURES.PLACEHOLDER)) {
137         element.value = placeholder;
138       }
139       return false;
140     }
141 
142     if (!checkTypeValidity_(element)) {
143       return false;
144     }
145 
146     /** @type {string} */ var pattern = element.getAttribute('pattern');
147     if (pattern && element.value) {
148       if (!(new RegExp(pattern, 'g')).test(element.value)) {
149         element.setCustomValidity(messages.getMessage('pattern'));
150         return false;
151       }
152     }
153 
154     element.setCustomValidity('');
155     if (!element.value && !forms.isSupported(forms.FEATURES.PLACEHOLDER)) {
156       element.value = placeholder;
157     }
158     return true;
159   }
160 
161   /**
162    * Returns whether a form will validate when it is submitted, without having
163    * to submit it.
164    * @param {Node|Element} form The form element.
165    * @return {boolean} Returns true if the form passes validation.
166    * @link http://msdn.microsoft.com/en-us/library/ie/hh772948
167    * @private
168    */
169   function checkFormValidity_(form) {
170     /** @type {Element} */ var invalid = dom.NULL;
171 
172     for (/** @type {number} */ var i = 0; i < form.elements.length; i++) {
173       /** @type {Element} */ var element = form.elements[i];
174 
175       if (element.checkValidity && !element.checkValidity()) {
176         if (!invalid)
177           invalid = element;
178       }
179 
180       if (!element.tooltip) {
181         element.tooltip = dom.createElement('SPAN');
182         dom.css.setClass(element.tooltip, 'validation-bubble');
183         element.parentNode.insertBefore(element.tooltip, element);
184         element.parentNode.insertBefore(element, element.tooltip);
185       }
186       element.tooltip.style.display = 'none';
187     }
188 
189     if (invalid) {
190       showValidationMessage_(invalid);
191       return false;
192     }
193     return true;
194   }
195 
196   /**
197    * @param {Node|Element} element HTML input or textarea element.
198    * @return {boolean} Returns true if element value is valid, false otherwise.
199    * @private
200    */
201   function checkTypeValidity_(element) {
202     /** @type {string} */ var value = element.value;
203     /** @type {string} */ var type = element.getAttribute('type');
204     /** @type {RegExp} */ var regexp = VALIDATORS[type];
205 
206     if (value && regexp && !regexp.test(value)) {
207       element.setCustomValidity(messages.getMessage(type));
208       return false;
209     }
210     return true;
211   }
212 
213   /**
214    * @param {Node|Element} element HTML element.
215    * @private
216    */
217   function showValidationMessage_(element) {
218     if (!showValidationMessage_.isAttached_) {
219       dom.events.addEventListener(
220           dom.document, dom.events.TYPE.MOUSEDOWN, function() {
221             if (invalidElement_ && invalidElement_.tooltip) {
222               invalidElement_.tooltip.style.display = 'none';
223             }
224           });
225 
226       showValidationMessage_.isAttached_ = true;
227     }
228     element.tooltip.innerHTML = element['validationMessage'];
229     element.tooltip.style.display = 'block';
230     element.focus();
231     invalidElement_ = element;
232   }
233 
234   /**
235    * @return {boolean} Returns true if browser supports HTML5 validation.
236    * @private
237    */
238   function isSupported_() {
239     return isSafari_() ? false : forms.isSupported(forms.FEATURES.VALIDATION);
240   }
241 
242   /**
243    * @return {boolean} Returns true if current browser is Safari.
244    * @private
245    */
246   function isSafari_() {
247     /** @type {boolean} */ var result = false;
248     // To the user Safari will behave no differently than a browser that doesn't
249     // support constraint validation at all.
250     if (window['webkitURL'] && window['HTMLElement']) {
251       if (window['HTMLElement'].toString().indexOf('Constructor') > 0) {
252         // window.HTMLElement.toString()
253         // chrome:
254         // "function HTMLElement() { [native code] }"
255         // safari (v. 3-8):
256         // "[object HTMLElementConstructor]"
257         result = true;
258       }
259     }
260     return result;
261   }
262 
263   /**
264    * @type {Node|Element}
265    * @private
266    */
267   var invalidElement_ = dom.NULL;
268 
269   /**
270    * @type {!locale.Validation}
271    * @private
272    */
273   var messages = new locale.Validation;
274 };
