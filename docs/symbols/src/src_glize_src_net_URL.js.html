 10 /**
 11  * Creates and return a URL object composed from the given parameters.
 12  * @param {string} url Is a String representing an absolute or relative URL.
 13  * @see https://developer.mozilla.org/en-US/docs/Web/API/URL
 14  * @see https://developer.mozilla.org/en-US/docs/Web/API/URL/URL
 15  * @constructor
 16  */
 17 net.URL = window['URL'] || window['webkitURL'] || function(url) {
 18 
 19   /**
 20    * @private
 21    */
 22   function init_() {
 23     self_['protocol'] = match_[2] ? match_[2] + ':' : '';
 24     self_['host'] = match_[4] || '';
 25     self_['hostname'] = self_['host'].split(':')[0];
 26     self_['port'] = +(self_['host'].split(':')[1]) || '';
 27     self_['pathname'] = match_[5] || '';
 28     self_['search'] = match_[7] ? ('?' + match_[7]) : '';
 29     self_['hash'] = match_[9] ? ('#' + match_[9]) : '';
 30 
 31     self_['toString'] = function() {
 32       return self_['protocol'] + '//' + self_['host'] +
 33              self_['pathname'] + self_['search'] + self_['hash'];
 34     };
 35   }
 36 
 37   /**
 38    * The regular expression for matching URL parts from <code>url</code> string.
 39    * @type {!RegExp}
 40    * @private
 41    */
 42   var regexp_ = /^(([^:\/?#]+):)?(\/\/([^\/?#]*))?([^?#]*)(\?([^#]*))?(#(.*))?/;
 43 
 44   /**
 45    * The list of matched URL parts.
 46    * @type {!Array.<string>}
 47    * @private
 48    */
 49   var match_ = (url || '').match(regexp_) || [];
 50 
 51   /**
 52    * The reference to current class instance.
 53    * Used in private methods and for preventing jslint errors.
 54    * @type {!net.URL}
 55    * @private
 56    */
 57   var self_ = this;
 58 
 59   init_();
 60 
 61   return self_;
 62 };
