 10 /**
 11  * Constructor of NumberFormatter.
 12  * @param {Object.<string, *>=} opt_options Formatter options.
 13  * @constructor
 14  * @example
 15  * options: {
 16  *   'decimal': '.', 'grouping': ',', 'fraction': 0, 'prefix': '', 'suffix': ''
 17  * }
 18  */
 19 formatters.NumberFormatter = function(opt_options) {
 20 
 21   /**
 22    * Formats given number according to given options.
 23    * @param {number} number The Number to be formatted.
 24    * @return {string} The formatted number string.
 25    * @example
 26    * <b>var</b> formatter = <b>new</b> formatters.NumberFormatter();
 27    * formatter.format(100);   // 100
 28    * formatter.format(1000);  // 1,000
 29    * formatter.format(1500);  // 1,500
 30    * formatter.format(10000); // 10,000
 31    * formatter.format(1e6);   // 1,000,000
 32    * <b>var</b> options = {'prefix': '$'};
 33    * <b>var</b> formatter = <b>new</b> formatters.NumberFormatter(options);
 34    * formatter.format(100);   // $100
 35    * formatter.format(1e6);   // $1,000,000
 36    */
 37   this.format = function(number) {
 38     /** @type {!Object.<string, *>} */ var options = getOptions_();
 39 
 40     /** @type {Array.<string>} */
 41     var result = (options['fraction'] ? number.toFixed(options['fraction']) :
 42                                         '' + number).split('.');
 43 
 44     return options['prefix'] +
 45         result[0].replace(/\B(?=(\d{3})+(?!\d))/g,
 46         /** @type {string} */ (options['grouping'])) +
 47         (result[1] ? options['decimal'] + result[1] : '') +
 48         options['suffix'];
 49   };
 50 
 51   /**
 52    * Rounds given number.
 53    * @param {number} number The Number to be rounded.
 54    * @return {string} The rounded number string.
 55    * @example
 56    * <b>var</b> formatter = <b>new</b> formatters.NumberFormatter();
 57    * formatter.round(100);   // 100
 58    * formatter.round(1000);  // 1k
 59    * formatter.round(1500);  // 1.5k
 60    * formatter.round(10000); // 10k
 61    * formatter.round(1e6);   // 10m
 62    * <b>var</b> options = {'prefix': '$'};
 63    * <b>var</b> formatter = <b>new</b> formatters.NumberFormatter(options);
 64    * formatter.round(100);   // $100
 65    * formatter.round(1e6);   // $10m
 66    */
 67   this.round = function(number) {
 68     // parseInt(number, 10) == ~~number
 69     /** @type {!Object.<string, *>} */ var options = getOptions_();
 70     /** @type {number} */
 71     var base = (~~number + '').length - (number < 0 ? 2 : 1);
 72     /** @type {number} */ var divider = 1;
 73     /** @type {number} */ var length = ~~(base / 3) * 3;
 74     for (/** @type {number} */ var i = 0; i < length; i++)
 75       divider *= 10;
 76 
 77     /** @type {number} */ var delta = number / divider;
 78     /** @type {number} */ var a = ~~delta;
 79     /** @type {string} */ var b = delta.toFixed(2 - base % 3);
 80 
 81     /** @type {string} */
 82     var result = (a == b ? a : b) + ['', 'k', 'm', 'g', 't'][~~(base / 3)];
 83     return options['prefix'] + result + options['suffix'];
 84   };
 85 
 86   /**
 87    * @return {!Object.<string, *>} Returns formatter options.
 88    * @private
 89    */
 90   function getOptions_() {
 91     opt_options = opt_options || {};
 92     for (/** @type {string} */ var key in defaults_) {
 93       if (!(key in opt_options)) {
 94         opt_options[key] = defaults_[key];
 95       }
 96     }
 97     return opt_options;
 98   }
 99 
100   /**
101    * Default formatting options.
102    * @dict
103    * @type {!Object.<string, *>}
104    * @private
105    */
106   var defaults_ = {
107     'decimal': '.',
108     'grouping': ',',
109     'fraction': 0,
110     'prefix': '',
111     'suffix': ''
112   };
113 };
