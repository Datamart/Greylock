 16 /**
 17  * @param {string} data Data content.
 18  * @param {string=} opt_fileName Optional file name.
 19  * @param {string=} opt_contentType Optional content type.
 20  * @static
 21  */
 22 util.FileUtils.saveAs = function(data, opt_fileName, opt_contentType) {
 23   /** @type {Element} */ var link = dom.createElement('A');
 24   opt_fileName = opt_fileName || 'file';
 25   opt_contentType = opt_contentType || 'text/plain';
 26   if (window.Blob && 'download' in link) {
 27     link['href'] = (window.URL || window.webkitURL).createObjectURL(
 28         new Blob([data],
 29         {'type': opt_contentType + ';charset=' + dom.CHARSET}));
 30     link.setAttribute('download', opt_fileName);
 31     dom.document.body.appendChild(link);
 32     link.click();
 33   } else {
 34     // @link http://msdn.microsoft.com/en-us/library/cc848897(VS.85).aspx
 35     window.open('data:' + opt_contentType +
 36                 ';base64,' + util.StringUtils.Base64.encode(data));
 37   }
 38 };
