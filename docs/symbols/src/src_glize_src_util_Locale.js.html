 11 /**
 12  * Construct a locale from language, country and variant.
 13  * @param {string} language The lowercase two-letter ISO-639 code.
 14  * @param {string=} opt_country The uppercase two-letter ISO-3166 code.
 15  * @param {string=} opt_variant The vendor and browser specific code.
 16  * @constructor
 17  * @link http://docs.oracle.com/javase/1.5.0/docs/api/java/util/Locale.html
 18  */
 19 util.Locale = function(language, opt_country, opt_variant) {
 20 
 21   /**
 22    * Returns the language code for this locale, which will either be the empty
 23    * string or a lowercase ISO-639 code.
 24    * @return {string} Returns the language code.
 25    */
 26   this.getLanguage = function() {
 27     return language || '';
 28   };
 29 
 30   /**
 31    * Returns the country/region code for this locale, which will either be the
 32    * empty string or an uppercase ISO-3166 2-letter code.
 33    * @return {string} Returns the country code.
 34    */
 35   this.getCountry = function() {
 36     return opt_country || '';
 37   };
 38 
 39   /**
 40    * Returns the variant code for this locale.
 41    * @return {string} Returns the variant code for this locale.
 42    */
 43   this.getVariant = function() {
 44     return opt_variant || '';
 45   };
 46 };
 47 
 48 
 49 /**
 50  * Gets the current value of the default locale.
 51  * @return {!util.Locale} Returns the default locale.
 52  * @static
 53  * @see docs.oracle.com/javase/1.5.0/docs/api/java/util/Locale.html#getDefault()
 54  */
 55 util.Locale.getDefault = function() {
 56   /** @type {string} */
 57   var language = navigator['userLanguage'] || navigator['language'];
 58   if (language) {
 59     return new util.Locale(language.substr(0, 2).toLowerCase());
 60   }
 61   return util.Locale.ENGLISH;
 62 };

 65 /**
 66  * Useful constant for language.
 67  * @type {!util.Locale}
 68  * @static
 69  * @see docs.oracle.com/javase/1.5.0/docs/api/java/util/Locale.html#ENGLISH
 70  */
 71 util.Locale.ENGLISH = new util.Locale('en');
