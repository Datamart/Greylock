 24 /**
 25  * Enum of HTML5 form features.
 26  * @enum {string}
 27  * @example <code>{
 28  *  PLACEHOLDER, VALIDATION
 29  * }</code>
 30  */
 31 forms.FEATURES = {
 32   PLACEHOLDER: 'placeholder',
 33   VALIDATION: 'required'
 34 };

 37 /**
 38  * @param {string} feature The html5 form feature.
 39  * @return {boolean} Returns true if feature is supported.
 40  * @see forms.FEATURES
 41  */
 42 forms.isSupported = function(feature) {
 43   forms.features_ = forms.features_ || {};
 44   if (!(feature in forms.features_)) {
 45     forms.features_[feature] = feature in dom.createElement('INPUT');
 46   }
 47   return forms.features_[feature];
 48 };

 51 /**
 52  * Initializes HTML5 form features: placeholder, validation, etc.
 53  * @param {Node} container The HTML container which contains form elements.
 54  * @static
 55  * @example
 56  * forms.init(document.forms[0]);
 57  */
 58 forms.init = function(container) {
 59   (new forms.PlaceHolder).init(container);
 60   (new forms.Validation).init(container);
 61 };

 64 /**
 65  * Adds onchange/oninput event handler on all form elements.
 66  * @param {HTMLFormElement} form Form element.
 67  * @param {!Function} handler Event handler.
 68  */
 69 forms.onchange = function(form, handler) {
 70 
 71   function checker_(element, handler) {
 72     /** @type {string} */ var value = element.value;
 73     setInterval(function() {
 74       if (value != element.value) {
 75         handler(/** @type {Event} */({'target': element}));
 76         value = element.value;
 77       }
 78     }, 99);
 79   }
 80 
 81   if (form) {
 82     /** @type {HTMLCollection} */ var elements = form.elements;
 83     /** @type {number} */ var length = elements.length;
 84     while (length) checker_(elements[--length], handler);
 85   }
 86 };
