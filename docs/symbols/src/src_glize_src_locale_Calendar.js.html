 10 /**
 11  * Constructor of Calendar.
 12  * @constructor
 13  */
 14 locale.Calendar = function() {
 15 
 16   /**
 17    * Gets locale week names.
 18    * @return {!Array.<string>} Returns week names.
 19    */
 20   this.getWeekNames = function() {
 21     return getData_('days');
 22   };
 23 
 24   /**
 25    * Gets locale month names.
 26    * @return {!Array.<string>} Returns month names.
 27    */
 28   this.getMonthNames = function() {
 29     return getData_('months');
 30   };
 31 
 32   /**
 33    * Gets locale month name.
 34    * @param {Date=} opt_date Optional date object, defaults is current date.
 35    * @return {string} Returns month name.
 36    * @see locale.Calendar#getMonthNames
 37    */
 38   this.getMonthName = function(opt_date) {
 39     opt_date = opt_date || new Date;
 40     return self_.getMonthNames()[opt_date.getMonth()];
 41   };
 42 
 43   /**
 44    * Gets month number by month name.
 45    * @param {string} name The month name.
 46    * @return {number} Returns month number.
 47    * @see locale.Calendar#getMonthNames
 48    */
 49   this.getMonthByName = function(name) {
 50     /** @type {!Array.<string>} */ var monthes = self_.getMonthNames();
 51     for (/** @type {number} */ var i = 0; i < monthes.length; i++) {
 52       if (name.substr(0, 3).toLowerCase() ==
 53           monthes[i].substr(0, 3).toLowerCase()) {
 54         return i;
 55       }
 56     }
 57     return -1;
 58   };
 59 
 60   /**
 61    * Gets locale data by key.
 62    * @param {string} key Data key.
 63    * @return {!Array.<string>} Returns locale data.
 64    * @private
 65    */
 66   function getData_(key) {
 67     return (data_[locale_.getLanguage()] ||
 68             data_[util.Locale.ENGLISH.getLanguage()])[key];
 69   }
 70 
 71   /**
 72    * The reference to current class instance. Used in private methods.
 73    * @type {!locale.Calendar}
 74    * @private
 75    */
 76   var self_ = this;
 77 
 78   /**
 79    * @type {!util.Locale}
 80    * @private
 81    */
 82   var locale_ = util.Locale.getDefault();
 83 
 84   /**
 85    * @type {!Object.<string, Object>}
 86    * @private
 87    */
 88   var data_ = {
 89     'en': {
 90       'days': ['Su', 'Mo', 'Tu', 'We', 'Th', 'Fr', 'Sa'],
 91       'months': ['January', 'February', 'March', 'April', 'May', 'June',
 92                  'July', 'August', 'September', 'October', 'November',
 93                  'December']
 94     },
 95     'de': {
 96       'days': ['So', 'Mo', 'Di', 'Mi', 'Do', 'Fr', 'Sa'],
 97       'months': ['Januar', 'Februar', 'März', 'April', 'Mai', 'Juni',
 98                  'Juli', 'August', 'September', 'Oktober', 'November',
 99                  'Dezember']
100     },
101     'ru': {
102       'days': ['Вс', 'Пн', 'Вт', 'Ср', 'Чт', 'Пт', 'Сб'],
103       'months': ['Январь', 'Февраль', 'Март', 'Апрель', 'Май', 'Июнь',
104                  'Июль', 'Август', 'Сентябрь', 'Октябрь', 'Ноябрь',
105                  'Декабрь']
106     }
107   };
108 };
