  9 /**
 10  * Simple implementation of HTML5 placeholder for non-supported browsers.
 11  * @constructor
 12  * @requires dom.events
 13  * @example
 14  * (<b>new</b> forms.PlaceHolder).init(document.forms[0]);
 15  */
 16 forms.PlaceHolder = function() {
 17 
 18   /**
 19    * Initializes HTML5 form placeholder feature.
 20    * @param {Node} container The HTML container which contains form elements.
 21    */
 22   this.init = function(container) {
 23     if (!forms.isSupported(forms.FEATURES.PLACEHOLDER)) {
 24       for (/** @type {number} */ var i = 0; i < forms.TAGS.length;) {
 25         initPlaceHolder_(dom.getElementsByTagName(container, forms.TAGS[i++]));
 26       }
 27     }
 28   };
 29 
 30   /**
 31    * Sets event listeners for elements with placeholder attribute.
 32    * @param {NodeList} elements Form elements.
 33    * @private
 34    */
 35   function initPlaceHolder_(elements) {
 36     for (/** @type {number} */ var i = 0; i < elements.length;) {
 37       /** @type {Node|Element} */ var element = elements[i++];
 38 
 39       if (element.getAttribute(attribute_)) {
 40         dom.events.addEventListener(
 41             element, dom.events.TYPE.FOCUS, handleEvent_);
 42         dom.events.addEventListener(
 43             element, dom.events.TYPE.BLUR, handleEvent_);
 44 
 45         handleEvent_({'target': element, 'type': dom.events.TYPE.BLUR});
 46       }
 47     }
 48   }
 49 
 50   /**
 51    * @param {Event|Object} e Event.
 52    * @private
 53    */
 54   function handleEvent_(e) {
 55     e = e || window.event;
 56     /** @type {Element} */ var element = e.srcElement || e.target;
 57     if (element) {
 58       dom.css.removeClass(element, attribute_);
 59       /** @type {string} */ var placeholder = element.getAttribute(attribute_);
 60       if (dom.events.TYPE.FOCUS == e.type && placeholder == element.value) {
 61         element.value = '';
 62       } else if (dom.events.TYPE.BLUR == e.type && '' == element.value) {
 63         element.value = placeholder;
 64         dom.css.addClass(element, attribute_);
 65       }
 66     }
 67   }
 68 
 69   /**
 70    * @type {string}
 71    * @private
 72    */
 73   var attribute_ = 'placeholder';
 74 };
