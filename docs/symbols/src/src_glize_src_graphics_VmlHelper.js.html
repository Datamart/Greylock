 19   /**
 20    * Converts SVG path string to VML path string.
 21    * @link https://raw.github.com/DmitryBaranovskiy/raphael/master/raphael.js
 22    * @param {string} svgPath The SVG path string.
 23    * @return {string} Returns converted VML path string.
 24    */
 25   this.getVmlPath = function(svgPath) {
 26     /** @type {!Object.<string, string>} */
 27     var map = {'M': 'm', 'L': 'l', 'C': 'c', 'Z': 'x',
 28       'm': 't', 'l': 'r', 'c': 'v', 'z': 'x'};
 29 
 30     /** @type {number} */ var zoom = 1; // 21600
 31     /** @type {Function} */
 32     var command = (/[ahqstv]/ig).test(svgPath) ? pathToCurve_ : pathToAbsolute_;
 33 
 34     if (command == pathToAbsolute_ && !(/[clmz]/g).test(svgPath)) {
 35       return svgPath.replace(/([clmz]),?([^clmz]*)/gi, function(mt, cmd, args) {
 36         /** @type {!Array.<number>} */ var values = [];
 37         /** @type {boolean} */ var isMove = cmd.toLowerCase() == 'm';
 38         /** @type {string} */ var str = map[cmd];
 39         args.replace(/-?[^,\s-]+/g, function(value) {
 40           if (isMove && values.length == 2) {
 41             str += values + map[cmd == 'm' ? 'l' : 'L'];
 42             values = [];
 43           }
 44           values.push(Math.round(value * zoom));
 45         });
 46         return str + values;
 47       });
 48     }
 49 
 50     /** @type {!Array} */ var pathArray = command(svgPath);
 51     /** @type {!Array} */ var result = [];
 52     for (/** @type {number} */ var i = 0; i < pathArray.length; i++) {
 53       /** @type {Array} */ var path = pathArray[i];
 54       /** @type {string} */ var cmd = path[0].toLowerCase();
 55       /** @type {number} */ var length = path.length;
 56       cmd = cmd == 'z' ? map[cmd] : cmd;
 57       for (/** @type {number} */ var j = 1; j < length; j++) {
 58         cmd += Math.round(path[j] * zoom) + (j != length - 1 ? ',' : '');
 59       }
 60       result.push(cmd);
 61     }
 62     return result.join(' ');
 63   };

 98   function quadraticToCurve_(x1, y1, ax, ay, x2, y2) {
 99     var _13 = 1 / 3,
100         _23 = 2 / 3;
101     return [
102       _13 * x1 + _23 * ax,
103       _13 * y1 + _23 * ay,
104       _13 * x2 + _23 * ax,
105       _13 * y2 + _23 * ay,
106       x2,
107       y2
108     ];

115   var pathToCurve_ = cacher_(function(path, path2) {
116     var pth = !path2 && paths_(path);
117     if (!path2 && pth.curve) {
118       return pathClone_(pth.curve);
119     }
120     var p = pathToAbsolute_(path),
121         p2 = path2 && pathToAbsolute_(path2),
122         attrs = {x: 0, y: 0, bx: 0, by: 0, X: 0, Y: 0, qx: null, qy: null},
123         attrs2 = {x: 0, y: 0, bx: 0, by: 0, X: 0, Y: 0, qx: null, qy: null};

125     function processPath(path, d) {
126       var nx, ny;
127       if (!path) {
128         return ['C', d.x, d.y, d.x, d.y, d.x, d.y];
129       }
130       !(path[0] in {T: 1, Q: 1}) && (d.qx = d.qy = null);
131       switch (path[0]) {
132         case 'M':
133           d.X = path[1];
134           d.Y = path[2];
135           break;
136         case 'A':
137           path = ['C'].concat(
138               arcToCurve_.apply(0, [d.x, d.y].concat(path.slice(1))));
139           break;
140         case 'S':
141           nx = d.x + (d.x - (d.bx || d.x));
142           ny = d.y + (d.y - (d.by || d.y));
143           path = ['C', nx, ny].concat(path.slice(1));
144           break;
145         case 'T':
146           d.qx = d.x + (d.x - (d.qx || d.x));
147           d.qy = d.y + (d.y - (d.qy || d.y));
148           path = ['C'].concat(
149               quadraticToCurve_(d.x, d.y, d.qx, d.qy, path[1], path[2]));
150           break;
151         case 'Q':
152           d.qx = path[1];
153           d.qy = path[2];
154           path = ['C'].concat(
155               quadraticToCurve_(d.x, d.y, path[1], path[2], path[3], path[4]));
156           break;
157         case 'L':
158           path = ['C'].concat(lineToCurve_(d.x, d.y, path[1], path[2]));
159           break;
160         case 'H':
161           path = ['C'].concat(lineToCurve_(d.x, d.y, path[1], d.y));
162           break;
163         case 'V':
164           path = ['C'].concat(lineToCurve_(d.x, d.y, d.x, path[1]));
165           break;
166         case 'Z':
167           path = ['C'].concat(lineToCurve_(d.x, d.y, d.X, d.Y));
168           break;
169       }
170       return path;
171     }

173     function fixArc(pp, i) {
174       if (pp[i].length > 7) {
175         pp[i].shift();
176         var pi = pp[i];
177         while (pi.length) {
178           pp.splice(i++, 0, ['C'].concat(pi.splice(0, 6)));
179         }
180         pp.splice(i, 1);
181         ii = Math.max(p.length, p2 && p2.length || 0);
182       }
183     }

185     function fixM(path1, path2, a1, a2, i) {
186       if (path1 && path2 && path1[i][0] == 'M' && path2[i][0] != 'M') {
187         path2.splice(i, 0, ['M', a2.x, a2.y]);
188         a1.bx = 0;
189         a1.by = 0;
190         a1.x = path1[i][1];
191         a1.y = path1[i][2];
192         ii = Math.max(p.length, p2 && p2.length || 0);
193       }
194     }
195 
196     for (var i = 0, ii = Math.max(p.length, p2 && p2.length || 0);
197         i < ii; i++) {
198       p[i] = processPath(p[i], attrs);
199       fixArc(p, i);
200       p2 && (p2[i] = processPath(p2[i], attrs2));
201       p2 && fixArc(p2, i);
202       fixM(p, p2, attrs, attrs2, i);
203       fixM(p2, p, attrs2, attrs, i);
204       var seg = p[i],
205               seg2 = p2 && p2[i];
206       /** @type {number} */ var seglen = seg.length;
207       /** @type {number} */ var seg2len = p2 ? seg2.length : 0;
208       attrs.x = seg[seglen - 2];
209       attrs.y = seg[seglen - 1];
210       attrs.bx = parseFloat(seg[seglen - 4]) || attrs.x;
211       attrs.by = parseFloat(seg[seglen - 3]) || attrs.y;
212       attrs2.bx = p2 && (parseFloat(seg2[seg2len - 4]) || attrs2.x);
213       attrs2.by = p2 && (parseFloat(seg2[seg2len - 3]) || attrs2.y);
214       attrs2.x = p2 && seg2[seg2len - 2];
215       attrs2.y = p2 && seg2[seg2len - 1];
216     }
217     if (!p2) {
218       pth.curve = pathClone_(p);
219     }
220     return p2 ? [p, p2] : p;
221   }, null, pathClone_);

223   /**
224    * @param {*} pathArray The SVG path string.
225    * @return {*} Returns converted path to curve.
226    * @private
227    */
228   function pathToAbsolute_(pathArray) {
229     var pth = paths_(pathArray);
230     if (pth.abs) {
231       return pathClone_(pth.abs);
232     }
233     if (!isArray_(pathArray) || !isArray_(pathArray && pathArray[0])) {
234       pathArray = parsePathString_(pathArray);
235     }
236     if (!pathArray || !pathArray.length) {
237       return [['M', 0, 0]];
238     }
239 
240     var res = [],
241         x = 0,
242         y = 0,
243         mx = 0,
244         my = 0,
245         start = 0;
246     if (pathArray[0][0] == 'M') {
247       x = +pathArray[0][1];
248       y = +pathArray[0][2];
249       mx = x;
250       my = y;
251       start++;
252       res[0] = ['M', x, y];
253     }
254     var crz = pathArray.length == 3 && pathArray[0][0] == 'M' &&
255               pathArray[1][0].toUpperCase() == 'R' &&
256               pathArray[2][0].toUpperCase() == 'Z';
257 
258     for (var r, pa, i = start, ii = pathArray.length; i < ii; i++) {
259       res.push(r = []);
260       pa = pathArray[i];
261       if (pa[0] != String.prototype.toUpperCase.call(pa[0])) {
262         r[0] = String.prototype.toUpperCase.call(pa[0]);
263         switch (r[0]) {
264           case 'A':
265             r[1] = pa[1];
266             r[2] = pa[2];
267             r[3] = pa[3];
268             r[4] = pa[4];
269             r[5] = pa[5];
270             r[6] = +(pa[6] + x);
271             r[7] = +(pa[7] + y);
272             break;
273           case 'V':
274             r[1] = +pa[1] + y;
275             break;
276           case 'H':
277             r[1] = +pa[1] + x;
278             break;
279           case 'R':
280             var dots = [x, y].concat(pa.slice(1));
281             for (var j = 2, jj = dots.length; j < jj; j++) {
282               dots[j] = +dots[j] + x;
283               dots[++j] = +dots[j] + y;
284             }
285             res.pop();
286             res = res.concat(catmullRom2bezier(dots, crz));
287             break;
288           case 'M':
289             mx = +pa[1] + x;
290             my = +pa[2] + y;
291           default:
292             for (j = 1, jj = pa.length; j < jj; j++) {
293               r[j] = +pa[j] + ((j % 2) ? x : y);
294             }
295         }
296       } else if (pa[0] == 'R') {
297         dots = [x, y].concat(pa.slice(1));
298         res.pop();
299         res = res.concat(catmullRom2bezier(dots, crz));
300         r = ['R'].concat(pa.slice(-2));
301       } else {
302         for (var k = 0, kk = pa.length; k < kk; k++) {
303           r[k] = pa[k];
304         }
305       }
306 
307       switch (r[0]) {
308         case 'Z':
309           x = mx;
310           y = my;
311           break;
312         case 'H':
313           x = r[1];
314           break;
315         case 'V':
316           y = r[1];
317           break;
318         case 'M':
319           mx = r[r.length - 2];
320           my = r[r.length - 1];
321         default:
322           x = r[r.length - 2];
323           y = r[r.length - 1];
324       }
325     }
326     res.toString = pathToString_;
327     pth.abs = pathClone_(res);
328     return res;
329   }

331   function paths_(ps) {
332     /** @type {!Object} */ var p = paths_.ps = paths_.ps || {};
333     if (p[ps]) {
334       p[ps].sleep = 100;
335     } else {
336       p[ps] = {sleep: 100};
337     }
338 
339     setTimeout(function() {
340       for (/** @type {string} */ var key in p) {
341         if (p.hasOwnProperty(key) && key != ps) {
342           p[key].sleep--;
343           !p[key].sleep && delete p[key];
344         }
345       }
346     }, 0);
347     return p[ps];
348   }

366   function clone_(obj) {
367     if (Object(obj) !== obj) {
368       return /** @type {Object} */ (obj);
369     }
370 
371     /** @type {Object} */ var result = new obj.constructor;
372     for (/** @type {string} */ var key in obj) {
373       if (obj.hasOwnProperty(key)) {
374         result[key] = clone_(obj[key]);
375       }
376     }
377     return result;

399   // replaced "\x09\x0a\x0b\x0c\x0d\x20\xa0\u1680\u180e\u2000\u2001\u2002" +
400   //          "\u2003\u2004\u2005\u2006\u2007\u2008\u2009\u200a\u202f\u205f" +
401   //          "\u3000\u2028\u2029"
402   // with "\s"
403   var pathCommand =
404       /([achlmrqstvz])[\s,]*((-?\d*\.?\d*(?:e[\-+]?\d+)?[\s]*,?[\s]*)+)/ig;
405   var pathValues = /(-?\d*\.?\d*(?:e[\-+]?\d+)?)[\s]*,?[\s]*/ig;

407   function parsePathString_(pathString) {
408     if (!pathString) {
409       return null;
410     }
411     var pth = paths_(pathString);
412     if (pth.arr) {
413       return pathClone_(pth.arr);
414     }
415 
416     var paramCounts = {'a': 7, 'c': 6, 'h': 1, 'l': 2, 'm': 2,
417       'r': 4, 'q': 4, 's': 4, 't': 2, 'v': 1, 'z': 0},
418         data = [];
419     if (isArray_(pathString) && isArray_(pathString[0])) {
420       data = pathClone_(pathString);
421     }
422     if (!data.length) {
423       pathString.replace(pathCommand, function(a, b, c) {
424         var params = [],
425             name = b.toLowerCase();
426         c.replace(pathValues, function(a, b) {
427           b && params.push(+b);
428         });
429         if (name == 'm' && params.length > 2) {
430           data.push([b].concat(params.splice(0, 2)));
431           name = 'l';
432           b = b == 'm' ? 'l' : 'L';
433         }
434         if (name == 'r') {
435           data.push([b].concat(params));
436         } else while (params.length >= paramCounts[name]) {
437           data.push([b].concat(params.splice(0, paramCounts[name])));
438           if (!paramCounts[name]) {
439             break;
440           }
441         }
442       });
443     }
444     data.toString = pathToString_;
445     pth.arr = pathClone_(data);
446     return data;
447   }

449   function catmullRom2bezier(crp, z) {
450     var d = [];
451     for (var i = 0, iLen = crp.length; iLen - 2 * !z > i; i += 2) {
452       var p = [
453         {x: +crp[i - 2], y: +crp[i - 1]},
454         {x: +crp[i], y: +crp[i + 1]},
455         {x: +crp[i + 2], y: +crp[i + 3]},
456         {x: +crp[i + 4], y: +crp[i + 5]}
457       ];
458       if (z) {
459         if (!i) {
460           p[0] = {x: +crp[iLen - 2], y: +crp[iLen - 1]};
461         } else if (iLen - 4 == i) {
462           p[3] = {x: +crp[0], y: +crp[1]};
463         } else if (iLen - 2 == i) {
464           p[2] = {x: +crp[0], y: +crp[1]};
465           p[3] = {x: +crp[2], y: +crp[3]};
466         }
467       } else {
468         if (iLen - 4 == i) {
469           p[3] = p[2];
470         } else if (!i) {
471           p[0] = {x: +crp[i], y: +crp[i + 1]};
472         }
473       }
474       d.push(['C',
475         (-p[0].x + 6 * p[1].x + p[2].x) / 6,
476         (-p[0].y + 6 * p[1].y + p[2].y) / 6,
477         (p[1].x + 6 * p[2].x - p[3].x) / 6,
478         (p[1].y + 6 * p[2].y - p[3].y) / 6,
479         p[2].x,
480         p[2].y
481       ]);
482     }
483 
484     return d;
485   }

487   function arcToCurve_(
488       x1, y1, rx, ry, angle, large_arc_flag, sweep_flag, x2, y2, recursive) {
489     // for more information of where this math came from visit:
490     // http://www.w3.org/TR/SVG11/implnote.html#ArcImplementationNotes
491     var _120 = Math.PI * 120 / 180,
492         rad = Math.PI / 180 * (+angle || 0),
493         res = [],
494         xy,
495         rotate = cacher_(function(x, y, rad) {
496           var X = x * Math.cos(rad) - y * Math.sin(rad),
497                 Y = x * Math.sin(rad) + y * Math.cos(rad);
498           return {x: X, y: Y};
499         });
500     if (!recursive) {
501       xy = rotate(x1, y1, -rad);
502       x1 = xy.x;
503       y1 = xy.y;
504       xy = rotate(x2, y2, -rad);
505       x2 = xy.x;
506       y2 = xy.y;
507       var cos = Math.cos(Math.PI / 180 * angle),
508           sin = Math.sin(Math.PI / 180 * angle),
509           x = (x1 - x2) / 2,
510           y = (y1 - y2) / 2;
511       var h = (x * x) / (rx * rx) + (y * y) / (ry * ry);
512       if (h > 1) {
513         h = Math.sqrt(h);
514         rx = h * rx;
515         ry = h * ry;
516       }
517       var rx2 = rx * rx,
518           ry2 = ry * ry,
519           k = (large_arc_flag == sweep_flag ? -1 : 1) *
520                 Math.sqrt(Math.abs((rx2 * ry2 - rx2 * y * y - ry2 * x * x) / (
521                   rx2 * y * y + ry2 * x * x))),
522           cx = k * rx * y / ry + (x1 + x2) / 2,
523           cy = k * -ry * x / rx + (y1 + y2) / 2,
524           f1 = Math.asin(((y1 - cy) / ry).toFixed(9)),
525           f2 = Math.asin(((y2 - cy) / ry).toFixed(9));
526 
527       f1 = x1 < cx ? Math.PI - f1 : f1;
528       f2 = x2 < cx ? Math.PI - f2 : f2;
529       f1 < 0 && (f1 = Math.PI * 2 + f1);
530       f2 < 0 && (f2 = Math.PI * 2 + f2);
531       if (sweep_flag && f1 > f2) {
532         f1 = f1 - Math.PI * 2;
533       }
534       if (!sweep_flag && f2 > f1) {
535         f2 = f2 - Math.PI * 2;
536       }
537     } else {
538       f1 = recursive[0];
539       f2 = recursive[1];
540       cx = recursive[2];
541       cy = recursive[3];
542     }
543     var df = f2 - f1;
544     if (Math.abs(df) > _120) {
545       var f2old = f2,
546           x2old = x2,
547           y2old = y2;
548       f2 = f1 + _120 * (sweep_flag && f2 > f1 ? 1 : -1);
549       x2 = cx + rx * Math.cos(f2);
550       y2 = cy + ry * Math.sin(f2);
551       res = arcToCurve_(x2, y2, rx, ry, angle, 0,
552           sweep_flag, x2old, y2old, [f2, f2old, cx, cy]);
553     }
554     df = f2 - f1;
555     var c1 = Math.cos(f1),
556         s1 = Math.sin(f1),
557         c2 = Math.cos(f2),
558         s2 = Math.sin(f2),
559         t = Math.tan(df / 4),
560         hx = 4 / 3 * rx * t,
561         hy = 4 / 3 * ry * t,
562         m1 = [x1, y1],
563         m2 = [x1 + hx * s1, y1 - hy * c1],
564         m3 = [x2 + hx * s2, y2 - hy * c2],
565         m4 = [x2, y2];
566     m2[0] = 2 * m1[0] - m2[0];
567     m2[1] = 2 * m1[1] - m2[1];
568     if (recursive) {
569       return [m2, m3, m4].concat(res);
570     } else {
571       res = [m2, m3, m4].concat(res).join().split(',');
572       var newres = [];
573       for (var i = 0, ii = res.length; i < ii; i++) {
574         newres[i] = i % 2 ? rotate(res[i - 1], res[i], rad).y :
575                             rotate(res[i], res[i + 1], rad).x;
576       }
577       return newres;
578     }
579   }
