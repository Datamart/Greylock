 10 /**
 11  * Constructor of InputRange.
 12  * @param {string|Element} input The input element or its ID attribute.
 13  * @param {Object=} opt_options A optional control's configuration options.
 14  * @constructor
 15  * @link http://www.w3.org/TR/html-markup/input.range.html
 16  */
 17 controls.html5.InputRange = function(input, opt_options) {
 18 
 19   /**
 20    * Defaults control options.
 21    * @dict
 22    * @see controls.html5.InputRange#getOptions_
 23    * @private
 24    */
 25   var defaults_ = {
 26     'input': {
 27       'border': 'inset 1px #ddd',
 28       'height': 1,
 29       'color': '#fff'
 30     },
 31     'point': {
 32       'border': 'solid 1px #333',
 33       'size' : 10,
 34       'color': '#eee'
 35     }
 36   };
 37 
 38   /**
 39    * @private
 40    */
 41   function init_() {
 42     if ('range' != input_.type) {
 43       options_ = getOptions_(opt_options);
 44 
 45       input_.style.border = options_['input']['border'];
 46       input_.style.height = options_['input']['height'] + 'px';
 47       input_.style.backgroundColor =
 48           input_.style.color = options_['input']['color'];
 49       getWrapper_().appendChild(getPoint_());
 50 
 51       dom.events.addEventListener(
 52           input_, dom.events.TYPE.MOUSEDOWN, mousedown_);
 53       dom.events.addEventListener(
 54           point_, dom.events.TYPE.MOUSEDOWN, mousedown_);
 55     }
 56   }
 57 
 58   /**
 59    * @return {Element} Returns reference to draggable element.
 60    * @private
 61    */
 62   function getPoint_() {
 63     if (!point_) {
 64       point_ = dom.createElement('DIV');
 65       point_.style.border = options_['point']['border'];
 66       point_.style.height = options_['point']['size'] + 'px';
 67       point_.style.width = options_['point']['size'] + 'px';
 68       point_.style.backgroundColor = options_['point']['color'];
 69       point_.style.position = 'absolute';
 70       point_.style.borderRadius = '50%';
 71       point_.style.top = '2px';
 72       point_.style.left = (input_.offsetWidth / 2) + 2 + 'px';
 73     }
 74     return point_;
 75   }
 76 
 77   /**
 78    * @return {Element} Returns reference to wrapper element.
 79    * @private
 80    */
 81   function getWrapper_() {
 82     if (!wrapper_) {
 83       wrapper_ = dom.createElement('SPAN');
 84       wrapper_.style.position = 'relative';
 85       dom.css.setClass(wrapper_, 'input-range-wrapper');
 86       input_.parentNode.insertBefore(wrapper_, input_);
 87       wrapper_.appendChild(input_);
 88     }
 89     return wrapper_;
 90   }
 91 
 92   /**
 93    * @param {Event} e The mousedown event.
 94    * @private
 95    */
 96   function mousedown_(e) {
 97     e = e || window.event;
 98     if ((e.target || e.srcElement) == point_) {
 99       dom.events.addEventListener(
100           dom.document, dom.events.TYPE.MOUSEMOVE, mousemove_);
101       dom.events.addEventListener(
102           dom.document, dom.events.TYPE.MOUSEUP, mouseup_);
103     }
104     mousemove_(e);
105   }
106 
107   /**
108    * @param {Event} e The mouseup event.
109    * @private
110    */
111   function mouseup_(e) {
112     dom.events.removeEventListener(
113         dom.document, dom.events.TYPE.MOUSEMOVE, mousemove_);
114     dom.events.removeEventListener(
115         dom.document, dom.events.TYPE.MOUSEUP, mouseup_);
116   }
117 
118   /**
119    * @param {Event} e The mousemove event.
120    * @private
121    */
122   function mousemove_(e) {
123     e = e || window.event;
124     /** @type {Object} */ var rect = dom.getBoundingClientRect(input_);
125     /** @type {number} */ var x = e.clientX;
126     if (x && x >= rect['left'] && x <= rect['right']) {
127       point_.style.left = x - rect['left'] + 'px';
128       /** @type {number} */ var value = +(input_.value || 0);
129       if (value >= min_ && value <= max_) {
130         // TODO: Increment / decrement value.
131         // TODO: Use step attribute.
132         input_.value = value;
133         dispatchEvents_();
134       }
135     }
136     // Prevent text selection.
137     dom.events.preventDefault(e);
138   }
139 
140   /**
141    * Dispatches input events.
142    * @return {boolean} Returns <code>true</code> if all events are
143    * dispatched successfully.
144    * @private
145    */
146   function dispatchEvents_() {
147     /** @type {boolean} */
148     var result = dom.events.dispatchEvent(input_, 'input');
149     result = dom.events.dispatchEvent(input_, dom.events.TYPE.CHANGE) && result;
150     return result;
151   }
152 
153   /**
154    * Gets control's options merged with defaults control's options.
155    * @param {Object=} opt_options A optional control's configuration options.
156    * @return {!Object.<string, *>} A map of name/value pairs.
157    * @private
158    */
159   function getOptions_(opt_options) {
160     return mergeOptions_(defaults_, opt_options || {});
161   }
162 
163   /**
164    * Merges control options with defaults options.
165    * @param {!Object.<string, *>} defaults Options map.
166    * @param {!Object.<string, *>} options Options map.
167    * @return {!Object.<string, *>} A map of key/value pairs of options.
168    * @private
169    */
170   function mergeOptions_(defaults, options) {
171     for (/** @type {string} */ var key in options) {
172       if (options[key] instanceof Array) {
173         defaults[key] = [].concat(options[key]);
174       } else if (typeof options[key] == 'object') {
175         defaults[key] = mergeOptions_(
176             /** @type {!Object.<string, *>} */ (defaults[key] || {}),
177             /** @type {!Object.<string, *>} */ (options[key]));
178       } else {
179         defaults[key] = options[key];
180       }
181     }
182     return /** @type {!Object.<string, *>} */ (defaults);
183   };
184 
185   /**
186    * The reference to wrapper element.
187    * @type {Element}
188    * @private
189    */
190   var wrapper_ = null;
191 
192   /**
193    * The reference to draggable element.
194    * @type {Element}
195    * @private
196    */
197   var point_ = null;
198 
199   /**
200    * The reference to input element.
201    * @type {Element}
202    * @private
203    */
204   var input_ = typeof input == 'string' ? dom.getElementById(input) : input;
205 
206   /**
207    * The expected lower bound for the input value.
208    * @type {number}
209    * @private
210    */
211   var min_ = +(input_.getAttribute('min') || -1e6);
212 
213   /**
214    * The expected upper bound for the input value.
215    * @type {number}
216    * @private
217    */
218   var max_ = +(input_.getAttribute('max') || 1e6);
219 
220   /**
221    * Specifies the value granularity of the input value.
222    * @type {number}
223    * @private
224    */
225   var step_ = +(input_.getAttribute('step') || 1);
226 
227   /**
228    * Storage for configuration options.
229    * @dict
230    * @private
231    */
232   var options_ = null;
233 
234   init_();
235 };
